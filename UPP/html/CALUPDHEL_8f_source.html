137  +uh(i-1,j-1,l)))*2.*r2dy
138  ENDIF
139 
140  updhel(i,j)=updhel(i,j)+(dvdx-dudy)*wh(i,j,l)*dz
141 
142  ENDIF

144  ENDDO
