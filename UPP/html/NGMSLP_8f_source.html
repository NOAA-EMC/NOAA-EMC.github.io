1  SUBROUTINE ngmslp
2 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 ! . . .
4 ! SUBPROGRAM: NGMSLP NMC SEA LEVEL PRESSURE REDUCTION
5 ! PRGRMMR: TREADON ORG: W/NP2 DATE: 93-02-02
6 !
7 ! ABSTRACT:
8 !
9 ! THIS ROUTINE COMPUTES SEA LEVEL PRESSURE USING THE
10 ! HYDROSTATIC EQUATION WITH THE SHUELL CORRECTION. THE
11 ! FOLLOWING IS BASED ON DOCUMENTATION IN SUBROUTINE
12 ! OUTHYDRO OF THE NGM:
13 !
14 ! THE FUNDAMENTAL HYDROSTATIC EQUATION IS
15 ! D(HEIGHT)
16 ! --------- = TAU = VIRTUAL TEMPERATURE * (RGAS/GRAVITY)
17 ! D (Z)
18 ! WHERE
19 ! Z = MINUS LOG OF PRESSURE (-LN(P)).
20 !
21 ! SEA-LEVEL PRESSURE IS COMPUTED FROM THE FORMULA
22 ! PRESS(MSL) = PRESS(GROUND) * EXP( F)
23 ! WHERE
24 ! F = HEIGHT OF GROUND / MEAN TAU
25 ! MEAN TAU = ( TAU(GRND) + TAU(SL) ) / 2
26 !
27 ! IN THE NGM TAU(GRND) AND TAU(SL) ARE FIRST SET USING A
28 ! 6.5DEG/KM LAPSE RATE FROM LOWEST MDL LEVEL. THIS IS MODIFIED
29 ! BY A CORRECTION BASED ON THE CRITICAL TAU OF THE SHUELL
30 ! CORRECTION:
31 ! TAUCR=(RGASD/GRAVITY) * 290.66
32 !
33 ! 1) WHERE ONLY TAU(SL) EXCEEDS TAUCR, CHANGE TAU(SL) TO TAUCR.
34 !
35 ! 2) WHERE BOTH TAU(SL) AND TAU(GRND) EXCEED TAUCR,
36 ! CHANGE TAU(SL) TO TAUCR-CONST*(TAU(GRND)-TAUCR )**2
37 ! WHERE CONST = .005 (GRAVITY/RGASD)
38 !
39 ! THE AVERAGE OF TAU(SL) AND TAU(GRND) IS THEN USED TOGETHER
40 ! WITH THE GROUND HEIGHT AND PRESSURE TO DERIVE THE PRESSURE
41 ! AT SEA LEVEL.
42 !
43 ! HEIGHT OF THE 1000MB SURFACE IS COMPUTED FROM THE MSL PRESSURE
44 ! FIELD USING THE FORMULA:
45 !
46 ! P(MSL) - P(1000MB) = MEAN DENSITY * GRAVITY * HGT(1000MBS)
47 !
48 ! WHERE P(MSL) IS THE SEA LEVEL PRESSURE FIELD WE HAVE JUST
49 ! COMPUTED.
50 !
51 !
52 ! MEB 6/13/02: THIS CODE HAS BEEN SIMPLIFIED CONSIDERABLY FROM
53 ! THE ONE USED IN ETAPOST. HORIZONTAL SMOOTHING HAS BEEN
54 ! REMOVED AND THE FIRST MODEL LEVEL IS USED RATHER
55 ! THAN THE MEAN OF THE VIRTUAL TEMPERATURES IN
56 ! THE LOWEST 30MB ABOVE GROUND TO COMPUTE TAU(GRND).
57 !
58 ! .
59 !
60 ! PROGRAM HISTORY LOG:

96  use params_mod, only: rd, gi, g, h1, d608, gamma, d50, p1000
97  use ctlblk_mod, only: jsta, jend, im, jm
98 !- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
99  implicit none
100 !
101  real,PARAMETER :: ZSL=0.0
102  real,PARAMETER :: TAUCR=rd*gi*290.66,const=0.005*g/rd
103  real,PARAMETER :: GORD=g/rd,dp=60.e2
104 !
105 ! DECLARE VARIABLES
106 !
107  integer I,J,LLMH
108  real ZSFC,PSFC,TVRT,TAU,TVRSFC,TAUSFC,TVRSL,TAUSL,TAUAVG, &
109  alpavg,pavg,rhoavg,rrhog
110 !
111 !**********************************************************************
112 ! START NGMSLP HERE.
113 !
114 ! LOOP OVER HORIZONTAL GRID.
115 !
116 !!$omp parallel do
117 !!$omp& private(llmh,pavg,psfc,qavg,rhoavg,rrhog,
118 !!$omp& tau,tauavg,tausfc,tausl,tavg,tvrbar,tvrsfc,tvrsl,
119 !!$omp& tvrt,tvrtal,zbar,zl,zsfc)
120  DO j=jsta,jend
121  DO i=1,im
122  llmh = nint(lmh(i,j))
123  zsfc = zint(i,j,llmh+1)
124  psfc = pint(i,j,llmh+1)
125  slp(i,j) = psfc
126 !
127 ! COMPUTE LAYER TAU (VIRTUAL TEMP*RD/G).
128  tvrt = t(i,j,llmh)*(h1+d608*q(i,j,llmh))
129  tau = tvrt*rd*gi
130 !
131 ! COMPUTE TAU AT THE GROUND (Z=ZSFC) AND SEA LEVEL (Z=0)
132 ! ASSUMING A CONSTANT LAPSE RATE OF GAMMA=6.5DEG/KM.
133 ! TVRSFC = TVRT + (ZSFC- ZSL)*GAMMA
134  tvrsfc = tvrt + (zmid(i,j,llmh) - zsfc)*gamma ! Chuang
135  tausfc = tvrsfc*rd*gi
136 ! TVRSL = TVRT + (ZSFC- ZSL)*GAMMA
137  tvrsl = tvrt + (zmid(i,j,llmh) - zsl)*gamma
138  tausl = tvrsl*rd*gi
139 !
140 ! IF NEED BE APPLY SHEULL CORRECTION.
141  IF ((tausl.GT.taucr).AND.(tausfc.LE.taucr)) THEN
142  tausl=taucr
143  ELSEIF ((tausl.GT.taucr).AND.(tausfc.GT.taucr)) THEN
144  tausl = taucr-const*(tausfc-taucr)**2
145  ENDIF
146 !
147 ! COMPUTE MEAN TAU.
148  tauavg = d50*(tausl+tausfc)
149 !
150 ! COMPUTE SEA LEVEL PRESSURE.
151  IF (abs(fis(i,j)).GT.1.0)slp(i,j) = psfc*exp(zsfc/tauavg)
152 !
153 ! COMPUTE 1000MB HEIGHTS.
154  alpavg = d50*(alog(psfc)+alog(slp(i,j)))
155  pavg = exp(alpavg)
156  rhoavg = pavg*gi/tauavg
157  rrhog = h1/(rhoavg*g)
158  z1000(i,j) = (slp(i,j)-p1000)*rrhog
159 !
160 ! MOVE TO NEXT HORIZONTAL GRIDPOINT.
161  ENDDO
162  ENDDO
163 !
164 !
165 ! END OF ROUTINE.
166 !
167 
168  RETURN
169  END
