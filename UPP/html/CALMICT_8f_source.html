108  zsmice=0. !--- Radar reflectivity from small ice

144  ENDIF
145  wv=q1d(i,j)/(1.-q1d(i,j))
146 !
147 !--- Saturation vapor pressure w/r/t water ( >=0C ) or ice ( <0C )
148 !
149  esat=1000.*fpvs(t1d(i,j))
150  qsat=eps*esat/(p1d(i,j)-esat)
151  rho=p1d(i,j)/(rd*t1d(i,j)*(1.+d608*q1d(i,j)))

179  ENDIF !--- End IF (QR1(I,J) .GT. EPSQ) block

188 ! -> Small ice particles are assumed to have a mean diameter of 50 microns.
189 ! * INDEXS - mean size of snow to the nearest micron (units of microns)
190 ! * RimeF - Rime Factor, which is the mass ratio of total (unrimed &
191 ! rimed) ice mass to the unrimed ice mass (>=1)
192 ! * QTICE - time-averaged mixing ratio of total ice

224  rqlice=rho*qlice
225  dum=xmimax*exp(.0536*tc)

232  nlimax=1.e3

251  ELSE IF (xli .LE. massi(450) ) THEN new_size

253  indexs=min(mdimax, max(mdimin, int(dli) ) )
254  ELSE IF (xli .LE. massi(mdimax) ) THEN new_size

256  indexs=min(mdimax, max(mdimin, int(dli) ) )

260  ENDIF new_size

262  nlice=rqlice/(rimef*massi(indexs)) !-- NLICE > NLImax
263  ENDIF no_hail
264  ENDIF new_nlice

266  !
267  !--- Equation (C.8) in Ferrier (1994, JAS, p. 272), which when
268  ! converted from cgs units to mks units results in the same
269  ! value for Cice, which is equal to the {} term below:
270  !
271  ! Zi={.224*720*(10**18)/[(PI*RHOL)**2]}*(RHO*QLICE)**2/NLICE1(I,J),
272  ! where RHOL=1000 kg/m**3 is the density of liquid water
273  !
274  !--- Valid only for exponential ice distributions
275  !
276  IF (nsmice > 0.) THEN
277  zsmice=cice*rho*rho*qsmice*qsmice/nsmice
278  ENDIF
279  if (nlice1(i,j) /= 0.0) zice=cice*rqlice*rqlice/nlice1(i,j)
280  IF (tc>=0.) zice=cwet*zice ! increased for wet ice
281  ENDIF ! End IF (QI1(I,J) .GT. 0.) THEN
282 !
283 !--- Assumed enhanced radar reflectivity when rain and ice coexist
284 ! above an assumed threshold mass content, RQmix
285 !
286 dbz_mix: IF (rqr>rqmix .AND. rqlice>rqmix) THEN
287  IF (rqr>rqlice) THEN
288  nmix=nrain
289  ELSE
290  nmix=nlice1(i,j)
291  ENDIF
292  dum=rqr+rqlice
293  zmix=cboth*dum*dum/nmix
294  IF (zmix > zrain+zice) THEN
295  IF (rqr>rqlice) THEN
296  zrain=zmix-zice
297  ELSE
298  zice=zmix-zrain
299  ENDIF
300  ENDIF
301  ENDIF dbz_mix
302 !
303 !--- Calculate total (convective + grid-scale) radar reflectivity
304 !
305 10 zice=zice+zsmice
306  ztot=zrain+zice+zconv
307  IF (ztot .GT. zmin) dbz1(i,j)= 10.*alog10(ztot)
308  IF (zrain .GT. zmin) dbzr1(i,j)=10.*alog10(zrain)
309  IF (zice .GT. zmin) dbzi1(i,j)=10.*alog10(zice)
310 ! IF (Zconv .GT. Zmin) DBZC1(I,J)=10.*ALOG10(Zsmice)
311  IF (zconv .GT. zmin) dbzc1(i,j)=10.*alog10(zconv)
312  ENDDO
313  ENDDO
314 !
315  RETURN
316  END

499 ! total ice into cloud ice & snow (precipitation ice)
500 !
501  IF (qi1(i,j) .GT. epsq) THEN
502  qice=qi1(i,j)
503  rho=p1d(i,j)/(rd*t1d(i,j)*(1.+oneps*q1d(i,j)))
504  rrho=1./rho
505  qsigrd=rhgrd*qsat
506  wvqw=wv+qw1(i,j)
507 !
508 ! * FLARGE - ratio of number of large ice to total (large & small) ice
509 ! * FSMALL - ratio of number of small ice crystals to large ice particles
510 ! -> Small ice particles are assumed to have a mean diameter of 50 microns.
511 ! * XSIMASS - used for calculating small ice mixing ratio
512 ! * XLIMASS - used for calculating large ice mixing ratio
513 ! * INDEXS - mean size of snow to the nearest micron (units of microns)
514 ! * RimeF - Rime Factor, which is the mass ratio of total (unrimed &
515 ! rimed) ice mass to the unrimed ice mass (>=1)
516 ! * FLIMASS - mass fraction of large ice
517 ! * QTICE - time-averaged mixing ratio of total ice
518 ! * QLICE - time-averaged mixing ratio of large ice
519 ! * NLICE1 - time-averaged number concentration of large ice
520 !
521  IF (tc.GE.0. .OR. wvqw.LT.qsigrd) THEN
522  flarge=1.
523  ELSE
524  flarge=flarge2 !-- specified in MICROINIT.f
525 !! IF (TC.GE.-8. .AND. TC.LE.-3.) FLARGE=.5*FLARGE
526  ENDIF
527  fsmall=(1.-flarge)/flarge
528  xsimass=rrho*massi(mdimin)*fsmall
529  dum=xmimax*exp(.0536*tc)
530  indexs=min(mdimax, max(mdimin, int(dum) ) )
531  rimef=amax1(1., fs1d(i,j) )
532  xlimass=rrho*rimef*massi(indexs)
533  flimass=xlimass/(xlimass+xsimass)
534  qlice=flimass*qice
535  nlice1(i,j)=qlice/xlimass
536  IF (nlice1(i,j).LT.nlimin .OR. nlice1(i,j).GT.nlimax) THEN
537 !
538 !--- Force NLICE1 to be between NLImin and NLImax
539 !
540  dum=max(nlimin, min(nlimax, nlice1(i,j)) )
541  xli=rho*(qice/dum-xsimass)/rimef
542  IF (xli .LE. massi(mdimin) ) THEN
543  indexs=mdimin
544  ELSE IF (xli .LE. massi(450) ) THEN
545  dli=9.5885e5*xli**.42066 ! DLI in microns
546  indexs=min(mdimax, max(mdimin, int(dli) ) )
547  ELSE IF (xli .LE. massi(mdimax) ) THEN
548  dli=3.9751e6*xli**.49870 ! DLI in microns
549  indexs=min(mdimax, max(mdimin, int(dli) ) )
550  ELSE
551  indexs=mdimax
552 !
553 !--- 8/22/01: Increase density of large ice if maximum limits
554 ! are reached for number concentration (NLImax) and mean size
555 ! (MDImax). Done to increase fall out of ice.
556 !
557  IF (dum .GE. nlimax) &
558  rimef=rho*(qice/nlimax-xsimass)/massi(indexs)
559  ENDIF ! End IF (XLI .LE. MASSI(MDImin) )
560  xlimass=rrho*rimef*massi(indexs)
561  flimass=xlimass/(xlimass+xsimass)
562  qlice=flimass*qice
563  nlice1(i,j)=qlice/xlimass
564  ENDIF ! End IF (NLICE.LT.NLImin ...
565  qs1(i,j)=amin1(qi1(i,j), qlice)
566  qi1(i,j)=amax1(0., qi1(i,j)-qs1(i,j))
567  !
568  !--- Equation (C.8) in Ferrier (1994, JAS, p. 272), which when
569  ! converted from cgs units to mks units results in the same
570  ! value for Cice, which is equal to the {} term below:
571  !
572  ! Zi={.224*720*(10**18)/[(PI*RHOL)**2]}*(RHO*QLICE)**2/NLICE1(I,J),
573  ! where RHOL=1000 kg/m**3 is the density of liquid water
574  !
575  !--- Valid only for exponential ice distributions
576  !
577  zice=cice*rho*rho*qlice*qlice/nlice1(i,j)
578  ENDIF ! End IF (QI1(I,J) .GT. 0.) THEN
579 !
580 !--- Calculate total (convective + grid-scale) radar reflectivity
581 10 ztot=zrain+zice+zconv
582  IF (ztot .GT. zmin) dbz1(i,j)= 10.*alog10(ztot)
583  IF (zrain .GT. zmin) dbzr1(i,j)=10.*alog10(zrain)
584  IF (zice .GT. zmin) dbzi1(i,j)=10.*alog10(zice)
585  IF (zconv .GT. zmin) dbzc1(i,j)=10.*alog10(zconv)
586  ENDDO
587  ENDDO
588 !
589  RETURN
590  END
