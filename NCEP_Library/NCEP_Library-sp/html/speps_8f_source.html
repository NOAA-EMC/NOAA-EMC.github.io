4 C
5 C SUBPROGRAM: SPEPS COMPUTE UTILITY SPECTRAL FIELDS
6 C PRGMMR: IREDELL ORG: W/NMC23 DATE: 92-10-31
7 C
8 C ABSTRACT: COMPUTES CONSTANT FIELDS INDEXED IN THE SPECTRAL DOMAIN
9 C IN "IBM ORDER" (ZONAL WAVENUMBER IS THE SLOWER INDEX).
10 C IF L IS THE ZONAL WAVENUMBER AND N IS THE TOTAL WAVENUMBER
11 C AND A IS THE EARTH RADIUS, THEN THE FIELDS RETURNED ARE:
12 C (1) NORMALIZING FACTOR EPSILON=SQRT((N**2-L**2)/(4*N**2-1))
13 C (2) LAPLACIAN FACTOR N*(N+1)/A**2
14 C (3) ZONAL DERIVATIVE/LAPLACIAN FACTOR L/(N*(N+1))*A
15 C (4) MERIDIONAL DERIVATIVE/LAPLACIAN FACTOR EPSILON/N*A
16 C
17 C PROGRAM HISTORY LOG:
18 C 91-10-31 MARK IREDELL
19 C
20 C USAGE: CALL SPEPS(I,M,EPS,EPSTOP,ENN1,ELONN1,EON,EONTOP)
21 C
22 C INPUT ARGUMENT LIST:
23 C I - INTEGER SPECTRAL DOMAIN SHAPE
24 C (0 FOR TRIANGULAR, 1 FOR RHOMBOIDAL)
25 C M - INTEGER SPECTRAL TRUNCATION
26 C
27 C OUTPUT ARGUMENT LIST:
28 C EPS - REAL ((M+1)*((I+1)*M+2)/2) SQRT((N**2-L**2)/(4*N**2-1))
29 C EPSTOP - REAL (M+1) SQRT((N**2-L**2)/(4*N**2-1)) OVER TOP
30 C ENN1 - REAL ((M+1)*((I+1)*M+2)/2) N*(N+1)/A**2
31 C ELONN1 - REAL ((M+1)*((I+1)*M+2)/2) L/(N*(N+1))*A
32 C EON - REAL ((M+1)*((I+1)*M+2)/2) EPSILON/N*A
33 C EONTOP - REAL (M+1) EPSILON/N*A OVER TOP
34 C
35 C ATTRIBUTES:
36 C LANGUAGE: CRAY FORTRAN
37 C
38 C$$$
39  REAL EPS((M+1)*((I+1)*M+2)/2),EPSTOP(M+1)
40  REAL ENN1((M+1)*((I+1)*M+2)/2),ELONN1((M+1)*((I+1)*M+2)/2)
41  REAL EON((M+1)*((I+1)*M+2)/2),EONTOP(M+1)
42  parameter(rerth=6.3712e6,ra2=1./rerth**2)
43 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
44  DO l=0,m
45  k=l*(2*m+(i-1)*(l-1))/2+l+1
46  eps(k)=0.
47  enn1(k)=ra2*l*(l+1)
48  elonn1(k)=rerth/(l+1)
49  eon(k)=0.
50  ENDDO
51  DO l=0,m
52  DO n=l+1,i*l+m
53  k=l*(2*m+(i-1)*(l-1))/2+n+1
54  eps(k)=sqrt(float(n**2-l**2)/float(4*n**2-1))
55  enn1(k)=ra2*n*(n+1)
56  elonn1(k)=rerth*l/(n*(n+1))
57  eon(k)=rerth/n*eps(k)
58  ENDDO
59  ENDDO
60  DO l=0,m
61  n=i*l+m+1
62  epstop(l+1)=sqrt(float(n**2-l**2)/float(4*n**2-1))
63  eontop(l+1)=rerth/n*epstop(l+1)
64  ENDDO
65 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
66  RETURN
67  END
