4 C
5 C SUBPROGRAM: SPGRADX COMPUTE X-GRADIENT IN FOURIER SPACE
6 C PRGMMR: IREDELL ORG: W/NMC23 DATE: 96-02-20
7 C
8 C ABSTRACT: THIS SUBPROGRAM COMPUTES THE X-GRADIENT OF FIELDS
9 C IN COMPLEX FOURIER SPACE.
10 C THE X-GRADIENT OF A VECTOR FIELD W IS
11 C WX=CONJG(W)*L/RERTH
12 C WHERE L IS THE WAVENUMBER AND RERTH IS THE EARTH RADIUS,
13 C SO THAT THE RESULT IS THE X-GRADIENT OF THE PSEUDO-VECTOR.
14 C THE X-GRADIENT OF A SCALAR FIELD W IS
15 C WX=CONJG(W)*L/(RERTH*CLAT)
16 C WHERE CLAT IS THE COSINE OF LATITUDE.
17 C AT THE POLE THIS IS UNDEFINED, SO THE WAY TO GET
18 C THE X-GRADIENT AT THE POLE IS BY PASSING BOTH
19 C THE WEIGHTED WAVENUMBER 0 AND THE UNWEIGHTED WAVENUMBER 1
20 C AMPLITUDES AT THE POLE AND SETTING MP=10.
21 C IN THIS CASE, THE WAVENUMBER 1 AMPLITUDES ARE USED
22 C TO COMPUTE THE X-GRADIENT AND THEN ZEROED OUT.
23 C
24 C PROGRAM HISTORY LOG:
25 C 1998-12-18 IREDELL
26 C
27 C USAGE: CALL SPGRADX(M,INCW,KMAX,W,WX)
28 C
29 C INPUT ARGUMENT LIST:
30 C M - INTEGER FOURIER WAVENUMBER TRUNCATION
31 C INCW - INTEGER FIRST DIMENSION OF THE COMPLEX AMPLITUDE ARRAY
32 C (INCW >= M+1)
33 C KMAX - INTEGER NUMBER OF FOURIER FIELDS
34 C MP - INTEGER (KM) IDENTIFIERS
35 C (0 OR 10 FOR SCALAR, 1 FOR VECTOR)
36 C CLAT - REAL COSINE OF LATITUDE
37 C W - COMPLEX(INCW,KMAX) FOURIER AMPLITUDES
38 C
39 C OUTPUT ARGUMENT LIST:
40 C W - COMPLEX(INCW,KMAX) FOURIER AMPLITUDES
41 C CORRECTED WHEN MP=10 AND CLAT=0
42 C WX - COMPLEX(INCW,KMAX) COMPLEX AMPLITUDES OF X-GRADIENTS
43 C
44 C SUBPROGRAMS CALLED:
45 C
46 C ATTRIBUTES:
47 C LANGUAGE: FORTRAN 90
48 C
49 C REMARKS:
50 C THIS SUBPROGRAM IS THREAD-SAFE.
51 C
52 C$$$
53  IMPLICIT NONE
54  INTEGER,INTENT(IN):: M,INCW,KMAX,MP(KMAX)
55  REAL,INTENT(IN):: CLAT
56  REAL,INTENT(INOUT):: W(2*INCW,KMAX)
57  REAL,INTENT(OUT):: WX(2*INCW,KMAX)

60 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
61  DO k=1,kmax
62  IF(mp(k).EQ.1) THEN
63  DO l=0,m
64  wx(2*l+1,k)=-w(2*l+2,k)*(l/rerth)
65  wx(2*l+2,k)=+w(2*l+1,k)*(l/rerth)
66  ENDDO
67  ELSEIF(clat.EQ.0.) THEN
68  DO l=0,m
69  wx(2*l+1,k)=0
70  wx(2*l+2,k)=0
71  ENDDO
72  IF(mp(k).EQ.10.AND.m.GE.2) THEN
73  wx(3,k)=-w(4,k)/rerth
74  wx(4,k)=+w(3,k)/rerth
75  w(3,k)=0
76  w(4,k)=0
77  ENDIF
78  ELSE
79  DO l=0,m
80  wx(2*l+1,k)=-w(2*l+2,k)*(l/(rerth*clat))
81  wx(2*l+2,k)=+w(2*l+1,k)*(l/(rerth*clat))
82  ENDDO
83  ENDIF
84  ENDDO
