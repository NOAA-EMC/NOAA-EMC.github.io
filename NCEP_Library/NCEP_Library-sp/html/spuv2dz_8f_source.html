3 C$$$ SUBPROGRAM DOCUMENTATION BLOCK
4 C
5 C SUBPROGRAM: SPUV2DZ COMPUTE DIVERGENCE AND VORTICITY FROM WINDS
6 C PRGMMR: IREDELL ORG: W/NMC23 DATE: 92-10-31
7 C
8 C ABSTRACT: COMPUTES THE DIVERGENCE AND VORTICITY FROM WIND COMPONENTS
9 C IN SPECTRAL SPACE.
10 C SUBPROGRAM SPEPS SHOULD BE CALLED ALREADY.
11 C IF L IS THE ZONAL WAVENUMBER, N IS THE TOTAL WAVENUMBER,
12 C EPS(L,N)=SQRT((N**2-L**2)/(4*N**2-1)) AND A IS EARTH RADIUS,
13 C THEN THE DIVERGENCE D IS COMPUTED AS
14 C D(L,N)=I*L*A*U(L,N)
15 C +EPS(L,N+1)*N*A*V(L,N+1)-EPS(L,N)*(N+1)*A*V(L,N-1)
16 C AND THE VORTICITY Z IS COMPUTED AS
17 C Z(L,N)=I*L*A*V(L,N)
18 C -EPS(L,N+1)*N*A*U(L,N+1)+EPS(L,N)*(N+1)*A*U(L,N-1)
19 C WHERE U IS THE ZONAL WIND AND V IS THE MERIDIONAL WIND.
20 C U AND V ARE WEIGHTED BY THE SECANT OF LATITUDE.
21 C EXTRA TERMS ARE USED OVER TOP OF THE SPECTRAL DOMAIN.
22 C ADVANTAGE IS TAKEN OF THE FACT THAT EPS(L,L)=0
23 C IN ORDER TO VECTORIZE OVER THE ENTIRE SPECTRAL DOMAIN.
24 C
25 C PROGRAM HISTORY LOG:
26 C 91-10-31 MARK IREDELL
27 C
28 C USAGE: CALL SPUV2DZ(I,M,ENN1,ELONN1,EON,EONTOP,U,V,UTOP,VTOP,D,Z)
29 C
30 C INPUT ARGUMENT LIST:
31 C I - INTEGER SPECTRAL DOMAIN SHAPE
32 C (0 FOR TRIANGULAR, 1 FOR RHOMBOIDAL)
33 C M - INTEGER SPECTRAL TRUNCATION
34 C ENN1 - REAL ((M+1)*((I+1)*M+2)/2) N*(N+1)/A**2
35 C ELONN1 - REAL ((M+1)*((I+1)*M+2)/2) L/(N*(N+1))*A
36 C EON - REAL ((M+1)*((I+1)*M+2)/2) EPSILON/N*A
37 C EONTOP - REAL (M+1) EPSILON/N*A OVER TOP
38 C U - REAL ((M+1)*((I+1)*M+2)) ZONAL WIND (OVER COSLAT)
39 C V - REAL ((M+1)*((I+1)*M+2)) MERID WIND (OVER COSLAT)
40 C UTOP - REAL (2*(M+1)) ZONAL WIND (OVER COSLAT) OVER TOP
41 C VTOP - REAL (2*(M+1)) MERID WIND (OVER COSLAT) OVER TOP
42 C
43 C OUTPUT ARGUMENT LIST:
44 C D - REAL ((M+1)*((I+1)*M+2)) DIVERGENCE
45 C Z - REAL ((M+1)*((I+1)*M+2)) VORTICITY
46 C
47 C ATTRIBUTES:
48 C LANGUAGE: CRAY FORTRAN
49 C
50 C$$$
51  REAL ENN1((M+1)*((I+1)*M+2)/2),ELONN1((M+1)*((I+1)*M+2)/2)
52  REAL EON((M+1)*((I+1)*M+2)/2),EONTOP(M+1)
53  REAL U((M+1)*((I+1)*M+2)),V((M+1)*((I+1)*M+2))
54  REAL UTOP(2*(M+1)),VTOP(2*(M+1))
55  REAL D((M+1)*((I+1)*M+2)),Z((M+1)*((I+1)*M+2))
56 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
57 C COMPUTE TERMS FROM THE SPECTRAL DOMAIN
58  k=1
59  d(2*k-1)=0.
60  d(2*k)=0.
61  z(2*k-1)=0.
62  z(2*k)=0.
63  DO k=2,(m+1)*((i+1)*m+2)/2-1
64  d(2*k-1)=-elonn1(k)*u(2*k)+eon(k+1)*v(2*k+1)-eon(k)*v(2*k-3)
65  d(2*k)=elonn1(k)*u(2*k-1)+eon(k+1)*v(2*k+2)-eon(k)*v(2*k-2)
66  z(2*k-1)=-elonn1(k)*v(2*k)-eon(k+1)*u(2*k+1)+eon(k)*u(2*k-3)
67  z(2*k)=elonn1(k)*v(2*k-1)-eon(k+1)*u(2*k+2)+eon(k)*u(2*k-2)
68  ENDDO
69  k=(m+1)*((i+1)*m+2)/2
70  d(2*k-1)=-elonn1(k)*u(2*k)-eon(k)*v(2*k-3)
71  d(2*k)=elonn1(k)*u(2*k-1)-eon(k)*v(2*k-2)
72  z(2*k-1)=-elonn1(k)*v(2*k)+eon(k)*u(2*k-3)
73  z(2*k)=elonn1(k)*v(2*k-1)+eon(k)*u(2*k-2)
74 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
75 C COMPUTE TERMS FROM OVER TOP OF THE SPECTRAL DOMAIN
76 CDIR$ IVDEP
77  DO l=0,m
78  k=l*(2*m+(i-1)*(l-1))/2+i*l+m+1
79  d(2*k-1)=d(2*k-1)+eontop(l+1)*vtop(2*l+1)
80  d(2*k)=d(2*k)+eontop(l+1)*vtop(2*l+2)
81  z(2*k-1)=z(2*k-1)-eontop(l+1)*utop(2*l+1)
82  z(2*k)=z(2*k)-eontop(l+1)*utop(2*l+2)
83  ENDDO
84 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
85 C MULTIPLY BY LAPLACIAN TERM
86  DO k=2,(m+1)*((i+1)*m+2)/2
87  d(2*k-1)=d(2*k-1)*enn1(k)
88  d(2*k)=d(2*k)*enn1(k)
89  z(2*k-1)=z(2*k-1)*enn1(k)
90  z(2*k)=z(2*k)*enn1(k)
91  ENDDO
92 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
93  RETURN
94  END
