7 C
8 C SUBPROGRAM: SPTRUNM SPECTRALLY INTERPOLATE SCALARS TO MERCATOR
9 C PRGMMR: IREDELL ORG: W/NMC23 DATE: 96-02-29
10 C
11 C ABSTRACT: THIS SUBPROGRAM SPECTRALLY TRUNCATES SCALAR FIELDS
12 C ON A GLOBAL CYLINDRICAL GRID, RETURNING THE FIELDS
13 C TO A MERCATOR GRID.
14 C THE WAVE-SPACE CAN BE EITHER TRIANGULAR OR RHOMBOIDAL.
15 C THE GRID-SPACE CAN BE EITHER AN EQUALLY-SPACED GRID
16 C (WITH OR WITHOUT POLE POINTS) OR A GAUSSIAN GRID.
17 C THE GRID FIELDS MAY HAVE GENERAL INDEXING.
18 C THE TRANSFORMS ARE ALL MULTIPROCESSED.
19 C TRANSFORM SEVERAL FIELDS AT A TIME TO IMPROVE VECTORIZATION.
20 C SUBPROGRAM CAN BE CALLED FROM A MULTIPROCESSING ENVIRONMENT.
21 C
22 C PROGRAM HISTORY LOG:
23 C 96-02-29 IREDELL
24 C
25 C USAGE: CALL SPTRUNM(IROMB,MAXWV,IDRTI,IMAXI,JMAXI,KMAX,MI,MJ,
26 C & IPRIME,ISKIPI,JSKIPI,KSKIPI,KGSKIP,
27 C & NISKIP,NJSKIP,JCPU,RLAT1,RLON1,DLAT,DLON,
28 C & GRIDI,GM)
29 C INPUT ARGUMENTS:
30 C IROMB - INTEGER SPECTRAL DOMAIN SHAPE
31 C (0 FOR TRIANGULAR, 1 FOR RHOMBOIDAL)
32 C MAXWV - INTEGER SPECTRAL TRUNCATION
33 C IDRTI - INTEGER INPUT GRID IDENTIFIER
34 C (IDRTI=4 FOR GAUSSIAN GRID,
35 C IDRTI=0 FOR EQUALLY-SPACED GRID INCLUDING POLES,
36 C IDRTI=256 FOR EQUALLY-SPACED GRID EXCLUDING POLES)
37 C IMAXI - INTEGER EVEN NUMBER OF INPUT LONGITUDES.
38 C JMAXI - INTEGER NUMBER OF INPUT LATITUDES.
39 C KMAX - INTEGER NUMBER OF FIELDS TO TRANSFORM.
40 C MI - INTEGER NUMBER OF POINTS IN THE FASTER ZONAL DIRECTION
41 C MJ - INTEGER NUMBER OF POINTS IN THE SLOWER MERID DIRECTION
42 C IPRIME - INTEGER INPUT LONGITUDE INDEX FOR THE PRIME MERIDIAN.
43 C (DEFAULTS TO 1 IF IPRIME=0)
44 C (OUTPUT LONGITUDE INDEX FOR PRIME MERIDIAN ASSUMED 1.)
45 C ISKIPI - INTEGER SKIP NUMBER BETWEEN INPUT LONGITUDES
46 C (DEFAULTS TO 1 IF ISKIPI=0)
47 C JSKIPI - INTEGER SKIP NUMBER BETWEEN INPUT LATITUDES FROM SOUTH
48 C (DEFAULTS TO -IMAXI IF JSKIPI=0)
49 C KSKIPI - INTEGER SKIP NUMBER BETWEEN INPUT GRID FIELDS
50 C (DEFAULTS TO IMAXI*JMAXI IF KSKIPI=0)
51 C KGSKIP - INTEGER SKIP NUMBER BETWEEN GRID FIELDS
52 C (DEFAULTS TO NPS*NPS IF KGSKIP=0)
53 C NISKIP - INTEGER SKIP NUMBER BETWEEN GRID I-POINTS
54 C (DEFAULTS TO 1 IF NISKIP=0)
55 C NJSKIP - INTEGER SKIP NUMBER BETWEEN GRID J-POINTS
56 C (DEFAULTS TO NPS IF NJSKIP=0)
57 C JCPU - INTEGER NUMBER OF CPUS OVER WHICH TO MULTIPROCESS
58 C (DEFAULTS TO ENVIRONMENT NCPUS IF JCPU=0)
59 C RLAT1 - REAL LATITUDE OF THE FIRST GRID POINT IN DEGREES
60 C RLON1 - REAL LONGITUDE OF THE FIRST GRID POINT IN DEGREES
61 C DLAT - REAL LATITUDE INCREMENT IN DEGREES SUCH THAT
62 C D(PHI)/D(J)=DLAT*COS(PHI) WHERE J IS MERIDIONAL INDEX.
63 C DLAT IS NEGATIVE FOR GRIDS INDEXED SOUTHWARD.
64 C (IN TERMS OF GRID INCREMENT DY VALID AT LATITUDE RLATI,
65 C THE LATITUDE INCREMENT DLAT IS DETERMINED AS
66 C DLAT=DPR*DY/(RERTH*COS(RLATI/DPR))
67 C WHERE DPR=180/PI AND RERTH IS EARTH'S RADIUS)
68 C DLON - REAL LONGITUDE INCREMENT IN DEGREES SUCH THAT
69 C D(LAMBDA)/D(I)=DLON WHERE I IS ZONAL INDEX.
70 C DLON IS NEGATIVE FOR GRIDS INDEXED WESTWARD.
71 C GRIDI - REAL (*) INPUT GRID FIELDS
72 C OUTPUT ARGUMENTS:
73 C GM - REAL (*) MERCATOR FIELDS
74 C
75 C SUBPROGRAMS CALLED:
76 C SPTRAN PERFORM A SCALAR SPHERICAL TRANSFORM
77 C SPTGPM TRANSFORM SPECTRAL SCALAR TO MERCATOR
78 C NCPUS GETS ENVIRONMENT NUMBER OF CPUS
79 C
80 C REMARKS: MINIMUM GRID DIMENSIONS FOR UNALIASED TRANSFORMS TO SPECTRAL:

113 C TRANSFORM WAVE TO OUTPUT
114  CALL sptgpm(iromb,maxwv,kmax,mi,mj,mdim,kgskip,niskip,njskip,
115  & rlat1,rlon1,dlat,dlon,w,gm)

117  END
