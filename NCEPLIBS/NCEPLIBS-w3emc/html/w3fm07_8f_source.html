4 C SUBR: W3FM07 - NINE-POINT SMOOTHER FOR RECTANGULAR GRIDS
5 C AUTHOR: CHASE, P. ORG: W345 DATE: APR 75
6 C
7 C ABSTRACT: SMOOTHS DATA ON A RECTANGULAR GRID USING A NINE-POINT
8 C SMOOTHING OPERATOR.
9 C
10 C PROGRAM HISTORY LOG:
11 C 75-04-01 P.CHASE

13 C 91-04-24 R.E.JONES CHANGE TO CRAY CFT77 FORTRAN
14 C
15 C USAGE: CALL W3FM07 (FIN, FOUT, CWORK, GAMMA, NCOL, NROW)
16 C
17 C INPUT:
18 C ' FIN' - REAL SIZE(NCOL*NROW) ARRAY OF DATA TO BE SMOOTHED
19 C ' CWORK' - REAL SIZE(2*NCOL*(NROW+2)) WORK ARRAY
20 C ' GAMMA' - COMPLEX SMOOTHING PARAMETER. THE IMAGINARY PART MUST
21 C BE POSITIVE.
22 C ' NCOL' - INTEGER NUMBER OF COLUMNS IN THE GRID
23 C ' NROW' - INTEGER NUMBER OF ROWS IN THE GRID
24 C OUTPUT:
25 C ' FOUT' - REAL SIZE(NCOL*NROW) ARRAY OF SMOOTHED DATA. MAY
26 C BE THE SAME ARRAY AS 'FIN' OR OVERLAP IT IN ANY
27 C FASHION.
28 C
29 C EXIT STATES: NONE
30 C
31 C EXTERNAL REFERENCES: NONE
32 C
33 C ATTRIBUTES:
34 C LANGUAGE: CRAY CFT77 FORTRAN
35 C MACHINE: CRAY Y-MP8/832
36 C
37 C$$$
38 C
39  REAL FIN(NCOL,NROW)
40  REAL FOUT(NCOL,NROW)
41 C
42  COMPLEX CWORK(NCOL,*),GAMMA,GAMMX,GAMA,GAMB,GAMC
43 C
44  gammx = gamma
45  xswtch = aimag(gammx)
46  ncolm = ncol-1
47  nrowm = nrow-1
48 C
49 C INITIALIZE WORK ARRAY. WORK ARRAY STARTS UP TWO ROWS SO IT CAN SMOOTH
50 C DOWNWARD WITHOUT OVERLAP OF SMOOTHED AND UNSMOOTHED DATA
51 C
52  DO 10 j = 1,nrow
53  DO 10 i = 1,ncol
54  cwork(i,j+2) = cmplx(fin(i,j),0.)
55  10 CONTINUE
56  IF (xswtch .EQ. 0.) GO TO 30
57  DO 20 j = 1,nrow,nrowm
58  jj = j+isign(1,nrowm-j)
59  DO 20 i = 1,ncol,ncolm
60  ii = i+isign(1,ncolm-i)
61  cwork(i,j+2) = cmplx(fin(i,jj)+fin(ii,j)-fin(ii,jj),0.)
62  20 CONTINUE
63 C
64 C SET SMOOTHING OPERATORS
65 C
66  30 gama = 0.50 * gammx * (1.0 - gammx)
67  gamb = 0.25 * gammx * gammx
68  gamc = 0.50 * gammx
69 C
70 C SMOOTH WORK ARRAY, PUTTING SMOOTHED POINTS DOWN TWO ROWS
71 C
72  cwork(1,1) = cwork(1,3)
73  cwork(ncol,1) = cwork(ncol,3)
74  DO 40 i = 2,ncolm
75  cwork(i,1) = cwork(i,3)+gamc*(cwork(i-1,3)-2.*cwork(i,3)+
76  & cwork(i+1,3))
77  40 CONTINUE
78  DO 60 j = 2,nrowm
79  DO 50 i = 1,ncol,ncolm
80  cwork(i,j) = cwork(i,j+2)+gamc*(cwork(i,j+1)-2.*cwork(i,j+2)+
81  & cwork(i,j+3))
82  50 CONTINUE
83  DO 60 i = 2,ncolm
84  cwork(i,j) = cwork(i,j+2)+gama*(cwork(i+1,j+2)+cwork(i-1,j+2)+
85  & cwork(i,j+1)+cwork(i,j+3)-4.*cwork(i,j+2))+gamb*(cwork(i-1,j+1)+
86  & cwork(i+1,j+1)+cwork(i-1,j+3)+cwork(i+1,j+3)-4.*cwork(i,j+2))
87  60 CONTINUE
88  cwork(1,nrow) = cwork(1,nrow+2)
89  cwork(ncol,nrow) = cwork(ncol,nrow+2)
90  DO 70 i = 2,ncolm
91  cwork(i,nrow) = cwork(i,nrow+2)+gamc*(cwork(i-1,nrow+2)-2.*
92  & cwork(i,nrow+2)+cwork(i+1,nrow+2))
93  70 CONTINUE
94 C
95 C IF IMAGINARY PART OF SMOOTHING PARAMETER IS NOT POSITIVE, DONE
96 C
97  IF (xswtch .LE. 0.) GO TO 90
98 C
99 C OTHERWISE MOVE WORK ARRAY BACK UP TWO ROWS
100 C
101  DO 80 jj=1,nrow
102  j = nrow+1-jj
103  DO 80 i=1,ncol
104  cwork(i,j+2) = cwork(i,j)
105  80 CONTINUE
106 C
107 C SET SMOOTHING PARAMETER FOR CONJUGATE PASS AND GO DO IT
108 C
109  gammx = conjg(gammx)
110  xswtch = aimag(gammx)
111  GO TO 30
112 C
113 C DONE. OUTPUT SMOOTH ARRAY
114 C
115  90 DO 100 j = 1,nrow
116  DO 100 i = 1,ncol
117  fout(i,j) = real(cwork(i,j))
118  100 CONTINUE
119  RETURN
120  END
