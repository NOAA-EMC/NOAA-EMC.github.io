2  SUBROUTINE sptgps(IROMB,MAXWV,KMAX,NPS,
3  & KWSKIP,KGSKIP,NISKIP,NJSKIP,
4  & TRUE,XMESH,ORIENT,WAVE,GN,GS)
5 C$$$ SUBPROGRAM DOCUMENTATION BLOCK
6 C
7 C SUBPROGRAM: SPTGPS TRANSFORM SPECTRAL SCALAR TO POLAR STEREO.
8 C PRGMMR: IREDELL ORG: W/NMC23 DATE: 96-02-29
9 C
10 C ABSTRACT: THIS SUBPROGRAM PERFORMS A SPHERICAL TRANSFORM
11 C FROM SPECTRAL COEFFICIENTS OF SCALAR QUANTITIES
12 C TO SCALAR FIELDS ON A PAIR OF POLAR STEREOGRAPHIC GRIDS.
13 C THE WAVE-SPACE CAN BE EITHER TRIANGULAR OR RHOMBOIDAL.
14 C THE WAVE AND GRID FIELDS MAY HAVE GENERAL INDEXING,
15 C BUT EACH WAVE FIELD IS IN SEQUENTIAL 'IBM ORDER',
16 C I.E. WITH ZONAL WAVENUMBER AS THE SLOWER INDEX.
17 C THE TWO SQUARE POLAR STEREOGRAPHIC GRIDS ARE CENTERED
18 C ON THE RESPECTIVE POLES, WITH THE ORIENTATION LONGITUDE
19 C OF THE SOUTHERN HEMISPHERE GRID 180 DEGREES OPPOSITE
20 C THAT OF THE NORTHERN HEMISPHERE GRID.
21 C
22 C THE TRANSFORM IS MADE EFFICIENT \ 4 | 5 /
23 C BY COMBINING POINTS IN EIGHT SECTORS \ | /
24 C OF EACH POLAR STEREOGRAPHIC GRID, 3 \ | / 6
25 C NUMBERED AS IN THE DIAGRAM AT RIGHT. \|/
26 C THE POLE AND THE SECTOR BOUNDARIES ----+----
27 C ARE TREATED SPECIALLY IN THE CODE. /|\
28 C UNFORTUNATELY, THIS APPROACH INDUCES 2 / | \ 7
29 C SOME HAIRY INDEXING AND CODE LOQUACITY, / | \
30 C FOR WHICH THE DEVELOPER APOLOGIZES. / 1 | 8 \
31 C
32 C THE TRANSFORMS ARE ALL MULTIPROCESSED OVER SECTOR POINTS.
33 C TRANSFORM SEVERAL FIELDS AT A TIME TO IMPROVE VECTORIZATION.
34 C SUBPROGRAM CAN BE CALLED FROM A MULTIPROCESSING ENVIRONMENT.
35 C
36 C PROGRAM HISTORY LOG:
37 C 96-02-29 IREDELL
38 C 1998-12-15 IREDELL OPENMP DIRECTIVES INSERTED
39 C
40 C USAGE: CALL SPTGPS(IROMB,MAXWV,KMAX,NPS,
41 C & KWSKIP,KGSKIP,NISKIP,NJSKIP,
42 C & TRUE,XMESH,ORIENT,WAVE,GN,GS)
43 C INPUT ARGUMENTS:
44 C IROMB - INTEGER SPECTRAL DOMAIN SHAPE
45 C (0 FOR TRIANGULAR, 1 FOR RHOMBOIDAL)
46 C MAXWV - INTEGER SPECTRAL TRUNCATION
47 C KMAX - INTEGER NUMBER OF FIELDS TO TRANSFORM.
48 C NPS - INTEGER ODD ORDER OF THE POLAR STEREOGRAPHIC GRIDS
49 C KWSKIP - INTEGER SKIP NUMBER BETWEEN WAVE FIELDS
50 C (DEFAULTS TO (MAXWV+1)*((IROMB+1)*MAXWV+2) IF KWSKIP=0)
51 C KGSKIP - INTEGER SKIP NUMBER BETWEEN GRID FIELDS
52 C (DEFAULTS TO NPS*NPS IF KGSKIP=0)
53 C NISKIP - INTEGER SKIP NUMBER BETWEEN GRID I-POINTS
54 C (DEFAULTS TO 1 IF NISKIP=0)
55 C NJSKIP - INTEGER SKIP NUMBER BETWEEN GRID J-POINTS
56 C (DEFAULTS TO NPS IF NJSKIP=0)
57 C TRUE - REAL LATITUDE AT WHICH PS GRID IS TRUE (USUALLY 60.)
58 C XMESH - REAL GRID LENGTH AT TRUE LATITUDE (M)
59 C ORIENT - REAL LONGITUDE AT BOTTOM OF NORTHERN PS GRID
60 C (SOUTHERN PS GRID WILL HAVE OPPOSITE ORIENTATION.)
61 C WAVE - REAL (*) WAVE FIELDS
62 C OUTPUT ARGUMENTS:
63 C GN - REAL (*) NORTHERN POLAR STEREOGRAPHIC FIELDS
64 C GS - REAL (*) SOUTHERN POLAR STEREOGRAPHIC FIELDS
65 C
66 C SUBPROGRAMS CALLED:
67 C SPWGET GET WAVE-SPACE CONSTANTS
68 C SPLEGEND COMPUTE LEGENDRE POLYNOMIALS
69 C SPSYNTH SYNTHESIZE FOURIER FROM SPECTRAL
70 C
71 C ATTRIBUTES:
72 C LANGUAGE: FORTRAN 77
73 C
74 C$$$
75  REAL WAVE(*),GN(*),GS(*)
76  REAL EPS((MAXWV+1)*((IROMB+1)*MAXWV+2)/2),EPSTOP(MAXWV+1)
77  REAL ENN1((MAXWV+1)*((IROMB+1)*MAXWV+2)/2)
78  REAL ELONN1((MAXWV+1)*((IROMB+1)*MAXWV+2)/2)
79  REAL EON((MAXWV+1)*((IROMB+1)*MAXWV+2)/2),EONTOP(MAXWV+1)
80  INTEGER MP(KMAX)
81  REAL SLON(MAXWV,8),CLON(MAXWV,8),SROT(0:3),CROT(0:3)
82  REAL WTOP(2*(MAXWV+1),KMAX)
83  REAL PLN((MAXWV+1)*((IROMB+1)*MAXWV+2)/2),PLNTOP(MAXWV+1)
84  REAL F(2*MAXWV+3,2,KMAX)
85  DATA srot/0.,1.,0.,-1./,crot/1.,0.,-1.,0./

88 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

93  idim=2*maxwv+3
94  kw=kwskip
95  kg=kgskip

98  IF(kw.EQ.0) kw=2*mx
99  IF(kg.EQ.0) kg=nps*nps

101  IF(nj.EQ.0) nj=nps
102  mp=0
103  nph=(nps-1)/2
104  gq=((1.+sin(true/dpr))*rerth/xmesh)**2
105 C$OMP PARALLEL DO
106  DO k=1,kmax
107  wtop(1:2*mxtop,k)=0
108  ENDDO
109 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
110 C CALCULATE POLE POINT
111  i1=nph+1
112  j1=nph+1
113  ij1=(i1-1)*ni+(j1-1)*nj+1

121  DO k=1,kmax
122  ijk1=ij1+(k-1)*kg

126 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
127 C CALCULATE POINTS ALONG THE ROW AND COLUMN OF THE POLE,

129 C$OMP PARALLEL DO PRIVATE(I1,J2,I2,J3,I3,J4,I4,J5,I5,J6,I6,J7,I7,J8,I8)

131 C$OMP& PRIVATE(IJK1,IJK2,IJK3,IJK4,IJK5,IJK6,IJK7,IJK8)

133 C$OMP& PRIVATE(PLN,PLNTOP,F,SLON,CLON,LR,LI)

135  i1=nph+1
136  radlon=orient/dpr
137  j3=nps+1-i1
138  i3=j1
139  j5=nps+1-j1
140  i5=nps+1-i1
141  j7=i1
142  i7=nps+1-j1

144  ij3=(i3-1)*ni+(j3-1)*nj+1

155  & clat1,pln,plntop,mp,wave,wtop,f)
156  DO l=1,maxwv
157  slon(l,1)=sin(l*radlon)
158  clon(l,1)=cos(l*radlon)
159  slon(l,3)=slon(l,1)*crot(mod(1*l,4))
160  & -clon(l,1)*srot(mod(1*l,4))
161  clon(l,3)=clon(l,1)*crot(mod(1*l,4))
162  & +slon(l,1)*srot(mod(1*l,4))
163  slon(l,5)=slon(l,1)*crot(mod(2*l,4))
164  & -clon(l,1)*srot(mod(2*l,4))
165  clon(l,5)=clon(l,1)*crot(mod(2*l,4))
166  & +slon(l,1)*srot(mod(2*l,4))
167  slon(l,7)=slon(l,1)*crot(mod(3*l,4))
168  & -clon(l,1)*srot(mod(3*l,4))
169  clon(l,7)=clon(l,1)*crot(mod(3*l,4))
170  & +slon(l,1)*srot(mod(3*l,4))

172 CDIR$ IVDEP
173  DO k=1,kmax

176  ijk5=ij5+(k-1)*kg
177  ijk7=ij7+(k-1)*kg
178  gn(ijk1)=f(1,1,k)
179  gn(ijk3)=f(1,1,k)
180  gn(ijk5)=f(1,1,k)
181  gn(ijk7)=f(1,1,k)
182  gs(ijk1)=f(1,2,k)
183  gs(ijk3)=f(1,2,k)
184  gs(ijk5)=f(1,2,k)
185  gs(ijk7)=f(1,2,k)
186  ENDDO
187  IF(kmax.EQ.1) THEN
188  DO l=1,maxwv
189  lr=2*l+1
190  li=2*l+2
191  gn(ij1)=gn(ij1)+2*(f(lr,1,1)*clon(l,1)
192  & -f(li,1,1)*slon(l,1))
193  gn(ij3)=gn(ij3)+2*(f(lr,1,1)*clon(l,3)
194  & -f(li,1,1)*slon(l,3))
195  gn(ij5)=gn(ij5)+2*(f(lr,1,1)*clon(l,5)
196  & -f(li,1,1)*slon(l,5))
197  gn(ij7)=gn(ij7)+2*(f(lr,1,1)*clon(l,7)
198  & -f(li,1,1)*slon(l,7))
199  gs(ij1)=gs(ij1)+2*(f(lr,2,1)*clon(l,5)
200  & -f(li,2,1)*slon(l,5))
201  gs(ij3)=gs(ij3)+2*(f(lr,2,1)*clon(l,3)
202  & -f(li,2,1)*slon(l,3))
203  gs(ij5)=gs(ij5)+2*(f(lr,2,1)*clon(l,1)
204  & -f(li,2,1)*slon(l,1))
205  gs(ij7)=gs(ij7)+2*(f(lr,2,1)*clon(l,7)
206  & -f(li,2,1)*slon(l,7))
207  ENDDO
208  ELSE
209  DO l=1,maxwv
210  lr=2*l+1
211  li=2*l+2
212 CDIR$ IVDEP
213  DO k=1,kmax
214  ijk1=ij1+(k-1)*kg
215  ijk3=ij3+(k-1)*kg
216  ijk5=ij5+(k-1)*kg
217  ijk7=ij7+(k-1)*kg
218  gn(ijk1)=gn(ijk1)+2*(f(lr,1,k)*clon(l,1)
219  & -f(li,1,k)*slon(l,1))
220  gn(ijk3)=gn(ijk3)+2*(f(lr,1,k)*clon(l,3)
221  & -f(li,1,k)*slon(l,3))
222  gn(ijk5)=gn(ijk5)+2*(f(lr,1,k)*clon(l,5)
223  & -f(li,1,k)*slon(l,5))
224  gn(ijk7)=gn(ijk7)+2*(f(lr,1,k)*clon(l,7)
225  & -f(li,1,k)*slon(l,7))
226  gs(ijk1)=gs(ijk1)+2*(f(lr,2,k)*clon(l,5)
227  & -f(li,2,k)*slon(l,5))
228  gs(ijk3)=gs(ijk3)+2*(f(lr,2,k)*clon(l,3)
229  & -f(li,2,k)*slon(l,3))
230  gs(ijk5)=gs(ijk5)+2*(f(lr,2,k)*clon(l,1)
231  & -f(li,2,k)*slon(l,1))
232  gs(ijk7)=gs(ijk7)+2*(f(lr,2,k)*clon(l,7)
233  & -f(li,2,k)*slon(l,7))
234  ENDDO
235  ENDDO
236  ENDIF
237  ENDDO
238 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
239 C CALCULATE POINTS ON THE MAIN DIAGONALS THROUGH THE POLE,
240 C STARTING CLOCKWISE OF THE ORIENTATION LONGITUDE AND GOING CLOCKWISE.
241 C$OMP PARALLEL DO PRIVATE(I1,J2,I2,J3,I3,J4,I4,J5,I5,J6,I6,J7,I7,J8,I8)
242 C$OMP& PRIVATE(IJ1,IJ2,IJ3,IJ4,IJ5,IJ6,IJ7,IJ8)
243 C$OMP& PRIVATE(IJK1,IJK2,IJK3,IJK4,IJK5,IJK6,IJK7,IJK8)
244 C$OMP& PRIVATE(DJ1,DI1,RQ,RADLON,RADLON1,RADLON2,SLAT1,CLAT1)
245 C$OMP& PRIVATE(PLN,PLNTOP,F,SLON,CLON,LR,LI)
246  DO j1=1,nph
247  i1=j1
248  radlon=(orient-45)/dpr
249  j3=nps+1-i1
250  i3=j1
251  j5=nps+1-j1
252  i5=nps+1-i1
253  j7=i1
254  i7=nps+1-j1
255  ij1=(i1-1)*ni+(j1-1)*nj+1
256  ij3=(i3-1)*ni+(j3-1)*nj+1
257  ij5=(i5-1)*ni+(j5-1)*nj+1

259  di1=i1-nph-1
260  dj1=j1-nph-1
261  rq=di1**2+dj1**2
262  slat1=(gq-rq)/(gq+rq)
263  clat1=sqrt(1.-slat1**2)
264  CALL splegend(iromb,maxwv,slat1,clat1,eps,epstop,
265  & pln,plntop)
266  CALL spsynth(iromb,maxwv,2*maxwv,idim,kw,2*mxtop,kmax,
267  & clat1,pln,plntop,mp,wave,wtop,f)
268  DO l=1,maxwv
269  slon(l,1)=sin(l*radlon)
270  clon(l,1)=cos(l*radlon)
271  slon(l,3)=slon(l,1)*crot(mod(1*l,4))
272  & -clon(l,1)*srot(mod(1*l,4))
273  clon(l,3)=clon(l,1)*crot(mod(1*l,4))
274  & +slon(l,1)*srot(mod(1*l,4))
275  slon(l,5)=slon(l,1)*crot(mod(2*l,4))
276  & -clon(l,1)*srot(mod(2*l,4))
277  clon(l,5)=clon(l,1)*crot(mod(2*l,4))
278  & +slon(l,1)*srot(mod(2*l,4))
279  slon(l,7)=slon(l,1)*crot(mod(3*l,4))
280  & -clon(l,1)*srot(mod(3*l,4))
281  clon(l,7)=clon(l,1)*crot(mod(3*l,4))
282  & +slon(l,1)*srot(mod(3*l,4))

284 CDIR$ IVDEP
285  DO k=1,kmax
286  ijk1=ij1+(k-1)*kg
287  ijk3=ij3+(k-1)*kg
288  ijk5=ij5+(k-1)*kg
289  ijk7=ij7+(k-1)*kg
290  gn(ijk1)=f(1,1,k)
291  gn(ijk3)=f(1,1,k)
292  gn(ijk5)=f(1,1,k)
293  gn(ijk7)=f(1,1,k)
294  gs(ijk1)=f(1,2,k)
295  gs(ijk3)=f(1,2,k)
296  gs(ijk5)=f(1,2,k)
297  gs(ijk7)=f(1,2,k)
298  ENDDO
299  IF(kmax.EQ.1) THEN
300  DO l=1,maxwv
301  lr=2*l+1
302  li=2*l+2
303  gn(ij1)=gn(ij1)+2*(f(lr,1,1)*clon(l,1)
304  & -f(li,1,1)*slon(l,1))
305  gn(ij3)=gn(ij3)+2*(f(lr,1,1)*clon(l,3)
306  & -f(li,1,1)*slon(l,3))
307  gn(ij5)=gn(ij5)+2*(f(lr,1,1)*clon(l,5)
308  & -f(li,1,1)*slon(l,5))
309  gn(ij7)=gn(ij7)+2*(f(lr,1,1)*clon(l,7)
310  & -f(li,1,1)*slon(l,7))
311  gs(ij1)=gs(ij1)+2*(f(lr,2,1)*clon(l,3)
312  & -f(li,2,1)*slon(l,3))
313  gs(ij3)=gs(ij3)+2*(f(lr,2,1)*clon(l,1)
314  & -f(li,2,1)*slon(l,1))
315  gs(ij5)=gs(ij5)+2*(f(lr,2,1)*clon(l,7)
316  & -f(li,2,1)*slon(l,7))
317  gs(ij7)=gs(ij7)+2*(f(lr,2,1)*clon(l,5)
318  & -f(li,2,1)*slon(l,5))
319  ENDDO
320  ELSE
321  DO l=1,maxwv
322  lr=2*l+1
323  li=2*l+2
324 CDIR$ IVDEP
325  DO k=1,kmax
326  ijk1=ij1+(k-1)*kg
327  ijk3=ij3+(k-1)*kg
328  ijk5=ij5+(k-1)*kg
329  ijk7=ij7+(k-1)*kg
330  gn(ijk1)=gn(ijk1)+2*(f(lr,1,k)*clon(l,1)
331  & -f(li,1,k)*slon(l,1))
332  gn(ijk3)=gn(ijk3)+2*(f(lr,1,k)*clon(l,3)
333  & -f(li,1,k)*slon(l,3))
334  gn(ijk5)=gn(ijk5)+2*(f(lr,1,k)*clon(l,5)
335  & -f(li,1,k)*slon(l,5))
336  gn(ijk7)=gn(ijk7)+2*(f(lr,1,k)*clon(l,7)
337  & -f(li,1,k)*slon(l,7))
338  gs(ijk1)=gs(ijk1)+2*(f(lr,2,k)*clon(l,3)
339  & -f(li,2,k)*slon(l,3))
340  gs(ijk3)=gs(ijk3)+2*(f(lr,2,k)*clon(l,1)
341  & -f(li,2,k)*slon(l,1))
342  gs(ijk5)=gs(ijk5)+2*(f(lr,2,k)*clon(l,7)
343  & -f(li,2,k)*slon(l,7))
344  gs(ijk7)=gs(ijk7)+2*(f(lr,2,k)*clon(l,5)
345  & -f(li,2,k)*slon(l,5))
346  ENDDO
347  ENDDO
348  ENDIF
349  ENDDO
350 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
351 C CALCULATE THE REMAINDER OF THE POLAR STEREOGRAPHIC DOMAIN,
352 C STARTING AT THE SECTOR JUST CLOCKWISE OF THE ORIENTATION LONGITUDE
353 C AND GOING CLOCKWISE UNTIL ALL EIGHT SECTORS ARE DONE.
354 C$OMP PARALLEL DO PRIVATE(I1,J2,I2,J3,I3,J4,I4,J5,I5,J6,I6,J7,I7,J8,I8)
355 C$OMP& PRIVATE(IJ1,IJ2,IJ3,IJ4,IJ5,IJ6,IJ7,IJ8)
356 C$OMP& PRIVATE(IJK1,IJK2,IJK3,IJK4,IJK5,IJK6,IJK7,IJK8)
357 C$OMP& PRIVATE(DJ1,DI1,RQ,RADLON,RADLON1,RADLON2,SLAT1,CLAT1)
358 C$OMP& PRIVATE(PLN,PLNTOP,F,SLON,CLON,LR,LI)
359  DO j1=1,nph-1
360  DO i1=j1+1,nph
361  j2=i1
362  i2=j1
363  j3=nps+1-i1
364  i3=j1

367  j5=nps+1-j1
368  i5=nps+1-i1
369  j6=nps+1-i1
370  i6=nps+1-j1
371  j7=i1
372  i7=nps+1-j1
373  j8=j1
374  i8=nps+1-i1
375  ij1=(i1-1)*ni+(j1-1)*nj+1
376  ij2=(i2-1)*ni+(j2-1)*nj+1

378  ij4=(i4-1)*ni+(j4-1)*nj+1
379  ij5=(i5-1)*ni+(j5-1)*nj+1
380  ij6=(i6-1)*ni+(j6-1)*nj+1
381  ij7=(i7-1)*ni+(j7-1)*nj+1
382  ij8=(i8-1)*ni+(j8-1)*nj+1
383  di1=i1-nph-1
384  dj1=j1-nph-1
385  rq=di1**2+dj1**2
386  slat1=(gq-rq)/(gq+rq)
387  clat1=sqrt(1.-slat1**2)
388  radlon1=orient/dpr+atan(-di1/dj1)
389  radlon2=(orient-45)/dpr*2-radlon1
390  CALL splegend(iromb,maxwv,slat1,clat1,eps,epstop,
391  & pln,plntop)
392  CALL spsynth(iromb,maxwv,2*maxwv,idim,kw,2*mxtop,kmax,
393  & clat1,pln,plntop,mp,wave,wtop,f)
394  DO l=1,maxwv
395  slon(l,1)=sin(l*radlon1)
396  clon(l,1)=cos(l*radlon1)
397  slon(l,2)=sin(l*radlon2)
398  clon(l,2)=cos(l*radlon2)
399  slon(l,3)=slon(l,1)*crot(mod(1*l,4))
