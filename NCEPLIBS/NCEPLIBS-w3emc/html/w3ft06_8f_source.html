1  SUBROUTINE w3ft06(ALOLA,APOLA,W1,W2,LINEAR)
2 C$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 C
4 C SUBPROGRAM: W3FT06 CONVERT (145,37) TO (65,65) S. HEMI. GRID
5 C AUTHOR: JONES,R.E. ORG: W342 DATE: 84-06-18
6 C
7 C ABSTRACT: CONVERT A SOUTHERN HEMISPHERE 2.5 DEGREE LAT.,LON. 145 BY
8 C 37 GRID TO A POLAR STEREOGRAPHIC 65 BY 65 GRID. THE POLAR
9 C STEREOGRAPHIC MAP PROJECTION IS TRUE AT 60 DEG. S. , THE MESH
10 C LENGTH IS 381 KM. AND THE ORIENTION IS 260 DEG. W.(100E) .
11 C
12 C PROGRAM HISTORY LOG:
13 C 84-06-18 R.E.JONES
14 C 91-07-30 R.E.JONES CONVERT TO CRAY CFT77 FORTRAN
15 C 92-05-02 R.E.JONES ADD SAVE
16 C
17 C USAGE: CALL W3FT06(ALOLA,APOLA,W1,W2,LINEAR)
18 C
19 C INPUT VARIABLES:
20 C NAMES INTERFACE DESCRIPTION OF VARIABLES AND TYPES
21 C ------ --------- -----------------------------------------------
22 C ALOLA ARG LIST 145*37 DEG 2.5 LAT,LON GRID S. HEMI.
23 C 5365 POINT GRID IS TYPE 30 OR 1E HEX O.N. 84
24 C LINEAR ARG LIST 1 LINEAR INTERPOLATION , NE.1 BIQUADRATIC
25 C
26 C OUTPUT VARIABLES:
27 C NAMES INTERFACE DESCRIPTION OF VARIABLES AND TYPES
28 C ------ --------- -----------------------------------------------
29 C APOLA ARG LIST 65*65 GRID OF SOUTHERN HEMI.
30 C 4225 POINT GRID IS TYPE 28 OR 1C HEX O.N. 84
31 C W1 ARG LIST 65*65 SCRATCH FIELD
32 C W2 ARG LIST 65*65 SCRATCH FIELD
33 C FT06F001 ERROR MESSAGE
34 C
35 C SUBPROGRAMS CALLED:
36 C NAMES LIBRARY
37 C ------------------------------------------------------- --------
38 C ASIN ATAN2 SYSTEM
39 C
40 C REMARKS:
41 C
42 C 1. W1 AND W2 ARE USED TO STORE SETS OF CONSTANTS WHICH ARE
43 C REUSABLE FOR REPEATED CALLS TO THE SUBROUTINE. IF THEY ARE
44 C OVER WRITTEN BY THE USER, A WARNING MESSAGE WILL BE PRINTED
45 C AND W1 AND W2 WILL BE RECOMPUTED.
46 C
47 C 2. WIND COMPONENTS ARE NOT ROTATED TO THE 65*65 GRID ORIENTATION
48 C AFTER INTERPOLATION. YOU MAY USE W3FC10 TO DO THIS.
49 C
50 C 3. THE GRID POINTS VALUES ON THE EQUATOR HAVE BEEN EXTRAPOLATED
51 C OUTWARD TO ALL THE GRID POINTS OUTSIDE THE EQUATOR ON THE 65*65
52 C GRID (ABOUT 1100 POINTS).
53 C
54 C 4. YOU SHOULD USE THE CRAY VECTORIZED VERION W3FT06V ON THE CRAY
55 C IT HAS 3 PARAMETERS IN THE CALL, RUNS ABOUT TIMES FASTER, USES
56 C MORE MEMORY.
57 C
58 C ATTRIBUTES:
59 C LANGUAGE: CRAY CFT77 FORTRAN
60 C MACHINE: CRAY Y-MP8/864
61 C
62 C$$$
63 C
64  REAL ALOLA(145,37)
65  REAL APOLA(4225)
66  REAL ERAS(4)
67  REAL SAVEW1(10)
68  REAL SAVEW2(10)
69  REAL W1(4225)
70  REAL W2(4225)

72  INTEGER JY(4)
73  INTEGER OUT
74 C
75  LOGICAL LIN
76 C
77  SAVE

79  DATA degprd/57.2957795/
80  DATA earthr/6371.2/
81  DATA iswt /0/
82  DATA out /6/
83 C
84  4000 FORMAT ( 52h *** warning , w1 or w2 scratch files over written ,,
85  & 43h i will restore them , burning up cpu time,,
86  & 14h in w3ft06 ***)
87 C
88  lin = .false.
89  IF (linear.EQ.1) lin = .true.
90  IF (iswt.EQ.0) GO TO 300
91 C
92 C TEST TO SEE IF W1 OR W2 WAS WRITTEN OVER
93 C
94  DO 100 kk=1,10
95  IF (savew1(kk).NE.w1(kk)) GO TO 200
96  IF (savew2(kk).NE.w2(kk)) GO TO 200
97  100 CONTINUE
98  GO TO 800
99 C
100  200 CONTINUE
101  WRITE (out,4000)
102 C
103  300 CONTINUE
104  deg = 2.5
105  nn = 0
106  xmesh = 381.0
107  gi2 = (1.86603*earthr) / xmesh
108  gi2 = gi2 * gi2
109 C
110 C DO LOOP 600 PUTS SUBROUTINE W3FB03 IN LINE
111 C
112  DO 600 j=1,65
113  xj = j - 33
114  xj2 = xj * xj
115  DO 600 i=1,65
116  xi = i - 33
117  r2 = xi*xi + xj2
118  IF (r2.NE.0.0) GO TO 400
119  wlon = 0.0
120  xlat = -90.0
121  GO TO 500
122  400 CONTINUE
123  xlong = degprd * atan2(xj,xi)
124  wlon = xlong -10.0
125  IF (wlon.LT.0.0) wlon = wlon + 360.0
126  xlat = asin((gi2-r2)/(gi2+r2))*degprd
127  xlat = -xlat
128  500 CONTINUE
129  xlat = xlat + 90.0
130  IF (wlon.GT.360.0) wlon = wlon - 360.0
131  IF (wlon.LT.0.0) wlon = wlon + 360.0
132  nn = nn + 1
133  w1(nn) = ( 360.0 - wlon ) / deg + 1.0
134  w2(nn) = xlat / deg + 1.0
135  600 CONTINUE
136 C
137  DO 700 kk=1,10
138  savew1(kk)=w1(kk)
139  savew2(kk)=w2(kk)
140  700 CONTINUE
141 C
142  iswt = 1
143 C
144  800 CONTINUE
145 C
146  DO 1900 kk=1,4225
147  i = w1(kk)
148  j = w2(kk)
149  fi = i
150  fj = j
151  xdeli = w1(kk) - fi
152  xdelj = w2(kk) - fj
153  ip1 = i + 1
154  jy(3) = j + 1
155  jy(2) = j
156  IF (lin) GO TO 900
157  ip2 = i + 2
158  im1 = i - 1
159  jy(4) = j + 2
160  jy(1) = j - 1
161  xi2tm = xdeli*(xdeli-1.)*.25
162  xj2tm = xdelj*(xdelj-1.)*.25
163  900 CONTINUE
164  IF ((i.LT.2).OR.(j.LT.2)) GO TO 1000
165  IF ((i.GT.142).OR.(j.GT.34)) GO TO 1000
166 C QUADRATIC (LINEAR TOO) OK W/O FURTHER ADO SO GO TO 1500
167  GO TO 1500
168 C
169  1000 CONTINUE
170  IF (i.EQ.1) GO TO 1100
171  IF (i.EQ.144) GO TO 1200
172  ip2 = i+2
173  im1 = i-1
174  GO TO 1300
175 C
176  1100 CONTINUE
177  ip2 = 3
178  im1 = 144
179  GO TO 1300
180 C
181  1200 CONTINUE
182  ip2 = 2
183  im1 = 143
184 C
185  1300 CONTINUE
186  ip1 = i + 1
187  IF (lin) GO TO 1400
188  IF ((j.LT.2).OR.(j.GE.36)) xj2tm=0.
189 C.....DO NOT ALLOW POINT OFF GRID,CYCLIC OR NOT
190  IF (ip2.LT.1) ip2 = 1
191  IF (im1.LT.1) im1 = 1
192  IF (ip2.GT.145) ip2 = 145
193  IF (im1.GT.145) im1 = 145
194 C
195  1400 CONTINUE
196 C.....DO NOT ALLOW POINT OFF GRID,CYCLIC OR NOT
197  IF (i.LT.1) i = 1
198  IF (ip1.LT.1) ip1 = 1
199  IF (i.GT.145) i = 145
200  IF (ip1.GT.145) ip1 = 145
201 C
202  1500 CONTINUE
203 C
204  IF (.NOT.lin) GO TO 1700
205 C
206 C LINEAR INTERPOLATION
207 C
208  DO 1600 k = 2,3
209  j1 = jy(k)
210  IF (j1.LT.1) j1=1
211  IF (j1.GT.37) j1=37
212  eras(k) = (alola(ip1,j1) - alola(i,j1)) * xdeli + alola(i,j1)
213  1600 CONTINUE
214 C
215  apola(kk) = eras(2) + (eras(3) - eras(2)) * xdelj
216  GO TO 1900
217 C
218  1700 CONTINUE
219 C
220 C QUADRATIC INTERPOLATION
221 C
222  DO 1800 k = 1,4
223  j1 = jy(k)
224 C.....DO NOT ALLOW POINT OFF GRID,CYCLIC OR NOT
225  IF (j1.LT.1) j1=1
226  IF (j1.GT.37) j1=37
227  eras(k)=(alola(ip1,j1)-alola(i,j1))*xdeli+alola(i,j1)+
228  & (alola(im1,j1)-alola(i,j1)-alola(ip1,j1)+
229  & alola(ip2,j1))*xi2tm
230  1800 CONTINUE
231 C
232  apola(kk) = eras(2)+(eras(3)-eras(2))*xdelj+(eras(1)-
233  & eras(2)-eras(3)+eras(4))*xj2tm
234 C
235  1900 CONTINUE
236 C
237 C SET POLE POINT, WMO STANDARD FOR U OR V
238 C
239  apola(2113) = alola(73,1)
240 C
241  RETURN
242  END
