21 C DATA MASKS /Z'00000001',Z'00000003',Z'00000007',Z'0000000F',
22 C & Z'0000001F',Z'0000003F',Z'0000007F',Z'000000FF',
23 C & Z'000001FF',Z'000003FF',Z'000007FF',Z'00000FFF',
24 C & Z'00001FFF',Z'00003FFF',Z'00007FFF',Z'0000FFFF',

26 C & Z'001FFFFF',Z'003FFFFF',Z'007FFFFF',Z'00FFFFFF',
27 C & Z'01FFFFFF',Z'03FFFFFF',Z'07FFFFFF',Z'0FFFFFFF',
28 C & Z'1FFFFFFF',Z'3FFFFFFF',Z'7FFFFFFF',Z'FFFFFFFF'/

30 C MASKS TABLE PUT IN DECIMAL SO IT WILL COMPILE ON ANY 32 BIT
31 C COMPUTER
32 C
33  DATA masks / 1, 3, 7, 15, 31, 63, 127, 255, 511, 1023, 2047,
34  & 4095, 8191, 16383, 32767, 65535, 131071, 262143, 524287,
35  & 1048575, 2097151, 4194303, 8388607, 16777215, 33554431,
36  & 67108863, 134217727, 268435455, 536870911, 1073741823,
37  & 2147483647, -1/
38 C
39 C NBYTE MUST BE LESS THAN OR EQUAL TO NBITSW
40 C
41  icon = nbitsw - nbyte
42  IF (icon.LT.0) RETURN
43  mask = masks(nbyte)
44 C
45 C INDEX TELLS HOW MANY WORDS INTO IOUT THE NEXT BYTE IS TO BE STORED.
46 C
47  index = ishft(iskip,-5)
48 C
49 C II TELLS HOW MANY BITS IN FROM THE LEFT SIDE OF THE WORD TO STORE IT.

52 C

54 C

56 C
57 C IWORDS TELLS HOW MANY WORDS TO SKIP FROM ONE BYTE TO THE NEXT.
58 C
59  iwords = istep / nbitsw
60 C
61 C IBITS TELLS HOW MANY BITS TO SKIP AFTER SKIPPING IWORDS.
62 C
63  ibits = mod(istep,nbitsw)
64 C
65  DO 10 i = 1,n
66  j = iand(mask,in(i))
67  movel = icon - ii
68 C
69 C BYTE IS TO BE STORED IN MIDDLE OF WORD. SHIFT LEFT.
70 C
71  IF (movel.GT.0) THEN
72  msk = ishft(mask,movel)
73  iout(index+1) = ior(iand(not(msk),iout(index+1)),
74  & ishft(j,movel))
75 C
76 C THE BYTE IS TO BE SPLIT ACROSS A WORD BREAK.
77 C
78  ELSE IF (movel.LT.0) THEN
79  msk = masks(nbyte+movel)
80  iout(index+1) = ior(iand(not(msk),iout(index+1)),
81  & ishft(j,movel))
82  itemp = iand(masks(nbitsw+movel),iout(index+2))
83  iout(index+2) = ior(itemp,ishft(j,nbitsw+movel))
84 C
85 C BYTE IS TO BE STORED RIGHT-ADJUSTED.
86 C
87  ELSE
88  iout(index+1) = ior(iand(not(mask),iout(index+1)),j)
89  ENDIF
90 C

92  index = index + iwords
93  IF (ii.GE.nbitsw) THEN
94  ii = ii - nbitsw
95  index = index + 1
96  ENDIF
