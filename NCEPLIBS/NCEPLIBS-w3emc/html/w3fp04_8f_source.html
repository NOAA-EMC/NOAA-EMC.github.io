4 C
5 C SUBPROGRAM: W3FP04 PRINT ARRAY OF DATA POINTS AT LAT/LON POINTS

8 C ABSTRACT: GIVEN AN ARRAY OF METEOROLOGICAL DATA AND CORRESPONDING
9 C LATITUDE/LONGITUDE POSITION FOR EACH DATA POINT, THESE DATA
10 C VALUES ARE PRINTED AT THEIR APPROXIMATE LATITUDE/LONGITUDE

12 C
13 C PROGRAM HISTORY LOG:
14 C 80-01-15 J. HORODECK

16 C 90-08-15 R.E.JONES CHANGE TO CRAY CFT77 FORTRAN

18 C USAGE: CALL W3FP04(IFLD,ALAT,ALON,TITLE,IDIM,CMIL,CMIR,CMJB,CMJT,
19 C INUM,XFAC,IERR)
20 C
21 C INPUT VARIABLES:
22 C NAMES INTERFACE DESCRIPTION OF VARIABLES AND TYPES

24 C IFLD ARG LIST REAL OR INTEGER FULLWORD ARRAY OF DATA POINTS

36 C CMJB ARG LIST REAL BOTTOM OF GRID - MINIMUM COARSE MESH
37 C J COORDINATE (MINIMUM VALUE OF 1.0)
38 C CMJT ARG LIST REAL TOP OF GRID - MAXIMUM COARSE MESH J
39 C COORDINATE (MAXIMUM VALUE OF 65.0)
40 C INUM ARG LIST INTEGER THREE DIGIT NUMBER FOR THE FOLLOWING:
41 C HUNDREDS DIGIT = TYPE OF DATA
42 C 1 = FIXED POINT
43 C 2 = FLOATING POINT
44 C 3 = ALPHANUMERIC
45 C TENS DIGIT = HEMISPHERIC REFERENCE
46 C 0 = NORTHERN HEMISPHERE
47 C 1 = SOUTHERN HEMISPHERE
48 C UNITS DIGIT = NUMBER OF CHARACTERS TO PLOT
49 C MINIMUM = 1 CHARACTER
50 C MAXIMUM = 4 CHARACTERS
51 C XFAC ARG LIST REAL MAP SCALE FACTOR (DESIRED MAP SCALE = XFAC
52 C * 1:30,000,000 (STANDARD NMC 65X65 GRID SCALE))
53 C
54 C OUTPUT VARIABLES:
55 C NAMES INTERFACE DESCRIPTION OF VARIABLES AND TYPES
56 C ------ --------- -----------------------------------------------
57 C TAPE6 PRINTED MAP WHEN EXECUTION IS SUCCESSFUL
58 C IERR ARG LIST INTEGER RETURN CODE
59 C
60 C SUBPROGRAMS CALLED:
61 C NAMES LIBRARY
62 C ------------------------------------------------------- --------
63 C W3FB04 W3FB05 W3AI15 W3LIB
64 C
65 C RESTRICTIONS: BECAUSE THIS CODE COULD PRODUCE CONSIDERABLE OUTPUT
66 C THE SUBSET OF THE NMC 65X65 GRID WHICH CAN BE PRINTED IS A
67 C FUNCTION OF THE MAP SCALE FACTOR, E.G. FOR XFAC=5 THE MAXIMUM
68 C RANGE OF I AND J IS 27.0, FOR XFAC=2 THE RANGE IS 64.0.
69 C
70 C ATTRIBUTES:
71 C LANGUAGE: CRAY CFT77 FORTRAN
72 C MACHINE: CRAY Y-MP8/832
73 C
74 C$$$

83 C
84  CHARACTER*1 KH(120,77), MEAN(4), KB, KM, KP, LC,
85  & kk(5,77,20), DATA(4), l1, l2, l3,
86  & ipole(4), kn, ks

88  CHARACTER*8 IFMTT

90  CHARACTER*28 FMT2
91  CHARACTER*24 FMT4
92 C
93  equivalence(mean(1),imean), (DATA(1),lfld)
94  equivalence(rfield,ifield)
95  equivalence(ifmtt,ifmt)
96 C
97  DATA jjaa /116/
98  DATA jjbb / 77/
99  DATA fmt1 /"(6X, ('+',I , X),//) "/
100  DATA fmt2 /"(' +',I3,1X, A1,' +',I3) "/

105 C
106  1001 FORMAT('1',16x,'PANEL #',i2,' OF ',i2,4x,10a8,/,/)
107  1003 FORMAT(6x,116a1)
108  2001 FORMAT(///,20x,'UPPER LEFT CORNER--LAT =',f6.2,' LON =',f7.2,'W'
109  & , 3x,'UPPER RIGHT CORNER--LAT =',f6.2,' LON =',f7.2,'W')
110  2002 FORMAT(20x,'LOWER LEFT CORNER--LAT =',f6.2,' LON =',f7.2,'W'
111  & , 3x,'LOWER RIGHT CORNER--LAT =',f6.2,' LON =',f7.2,'W')
112  2003 FORMAT(/,/, 16x, 'PANEL #', i2, ' OF ', i2, 4x, 10a8)
113  9001 FORMAT(/,5x,'CMIL = ',f8.1,' CMIR = ',f8.1,' HIGH AND LOW'
114  & ,' VALUES REVERSED......RETURN......')
115  9002 FORMAT(/,5x,'CMJB = ',f8.1,' CMJT = ',f8.1,' HIGH AND LOW'
116  & ,' VALUES REVERSED......RETURN......')
117  9003 FORMAT(/,5x,f8.1,' IS ILLEGAL VALUE FOR LOW I. IT IS NOW 1.0')
118  9004 FORMAT(/,5x,f8.1,' IS ILLEGAL VALUE FOR HIGH I. IT IS NOW 65.0')
119  9005 FORMAT(/,5x,f8.1,' IS ILLEGAL VALUE FOR LOW J. IT IS NOW 1.0')
120  9006 FORMAT(/,5x,f8.1,' IS ILLEGAL VALUE FOR HIGH J. IT IS NOW 65.0')
121  9007 FORMAT(/,5x,'REQUESTED NUMBER OF CHARACTERS TO PLOT(',i2,' )IS'
122  & ,' NOT ALLOWED. FOUR(4) IS MAXIMUM. THATS ALL YOU GET')
123  9008 FORMAT(/,5x,'REQUESTED SUBSET OF 65X65 GRID CANNOT CURRENTLY '
124  & ,'BE PLOTTED WITH MAP SCALE FACTOR',f5.1,/5x,'IF PLOT '
125  & ,'IS NECESSARY, CONTACT JOHN M. HORODECK,ESQ. NMC/DD'
126  & ,'/SEB FOR ASSISTANCE')
127  9009 FORMAT(/,5x,i4,' IS INVALID HEMISPHERIC REFERENCE'
128  & , '......RETURN......')
129  9010 FORMAT(/,5x,'HUNDREDS DIGIT OF INUM(INUM =',i4,') IS'
130  & , ' INVALID......RETURN......')
131 C
132 C TEST I,J VALUES FOR RANGE AND ORDER
133 C
134  IF (cmir.GT.cmil) GO TO 1
135  ierr = 1
136  print 9001, cmil, cmir
137  RETURN
138  1 CONTINUE
139  IF (cmjt.GT.cmjb) GO TO 2
140  ierr = 1
141  print 9002, cmjb, cmjt
142  RETURN
143  2 CONTINUE
144  IF (cmil.GE.1.0) GO TO 3
145  print 9003, cmil
146  cmil = 1.0
147  3 CONTINUE
148  IF (cmir.LE.65.0) GO TO 4
149  print 9004, cmir
150  cmir = 65.0
151  4 CONTINUE
152  IF (cmjb.GE.1.0) GO TO 5
153  print 9005, cmjb
154  cmjb = 1.0
155  5 CONTINUE
156  IF (cmjt.LE.65.0) GO TO 6
157  print 9006, cmjt
158  cmjt = 65.0
159  6 CONTINUE
160 C
161 C CALCULATE VARIOUS LIMITS
162 C
163  lnum = mod(inum,10)
164  nref = (mod(inum,100))/10
165 C
166 C TEST FOR INCORRECT ARGUMENTS PASSED
167 C
168  IF (lnum.LE.4) GO TO 7
169  print 9007, lnum
170  lnum = 4
171  7 CONTINUE
172  IF (nref.LE.1) GO TO 8
173  ierr = 1
174  print 9009, nref
175  RETURN
176  8 CONTINUE
177  IF ((inum/100).LE.3) GO TO 81
178  ierr = 1
179  print 9010, inum
180  RETURN
181  81 CONTINUE
182 C
183  lnump1 = lnum + 1
184  i1 = (cmil-1.0)*xfac + 1.0
185  i2 = (cmir-1.0)*xfac + 1.0
186  j1 = (cmjb-1.0)*xfac + 1.0
187  j2 = (cmjt-1.0)*xfac + 1.0
188 C
189 C WILL THIS PLOT BE TOO LARGE?
190 C
191  IF (((i2-i1).LT.139).AND.((j2-j1).LT.139)) GO TO 9
192  ierr = 1
193  print 9008, xfac
194  RETURN
195  9 CONTINUE
196 C
197  offi = i1 - 1
198  offj = j1 - 1
199  jja = (i2-i1)*5 + 1
200  jjb = (j2-j1)*4 + 1
201  jjam1 = jja - 1
202  jjbbm1 = jjbb - 1
203  jjaam1 = jjaa - 1
204  jjaapn = jjaa + lnum
205  ibegin = lnump1 + 1
206  ipage = (jjam1/jjaa) + 1
207  jpage = (jjb/jjbb) + 1
208  xmesh = 381.0/xfac
209  xip = 32.0*xfac + 1.0
210  xjp = 32.0*xfac + 1.0
211  iixip = (xip-offi)*5 - 4
212  jjxjp = (xjp-offj)*4 - 3
213 C
214 C PLOT DATA ONE PANEL AT A TIME IN SECTIONS
215 C
216  DO 150 nx=1,ipage
217  a = .false.
218 C
219 C SET LIMITS OF I TO BE PRINTED
220 C
221  il(nx) = i1 + (23*(nx-1))
222  IF (nx.NE.ipage) ir(nx) = i1 + (23*nx)
223  IF (nx.EQ.ipage) ir(nx) = i2
224  imax = ir(nx) - offi
225  imin = il(nx) - offi
226  m = 0
227 C
228 C FILL ARRAY WITH VALUES OF I TO BE PRINTED AT TOP OF PAGE
229 C
230  DO 10 i = imin,imax
231  m = m + 1
232  line(m) = i
233  10 CONTINUE
234 C
235 C CALCULATE WIDTH OF PANEL IN INTEGERS AND
236 C CHARACTERS FROM WHICH DETERMINE FORMAT
237 C FIELD COUNT AND CONVERT BINARY TO ASCII
238 C
239 C PRINT TOP LINE OF I
240 C
241  la = (imax-imin) + 1
242  mmm = (la*5) - 4
243  IF (la.EQ.24) GO TO 13
244  CALL w3ai15(la,ifmt,1,4,kp)
245  fmt1(5:8) = ifmtt(1:4)
246  fmt4(9:12) = ifmtt(1:4)
247  CALL w3ai15(mmm,ifmt,1,4,kp)
248  fmt2(13:16) = ifmtt(1:4)
249  GO TO 16
250  13 CONTINUE
251  fmt1(5:8) = l24
252  fmt2(13:16) = l116
253  fmt4(9:12) = l24
254  16 CONTINUE
255  IF (la.LT.100) GO TO 19
256  fmt1(15:15) = l3
257  fmt1(17:17) = l1
258  fmt4(19:19) = l3
259  fmt4(21:21) = l1
260  GO TO 22
261  19 CONTINUE
262  fmt1(15:15) = l2
263  fmt1(17:17) = l2
264  fmt4(19:19) = l2
265  fmt4(21:21) = l2
266  22 CONTINUE
267  print 1001, nx, ipage, title
268  WRITE(6,fmt1) (line(n), n=1,la)
269 C
270 C PREPARE TO PRINT SECTIONS OF EACH PANEL
271 C
272  DO 140 jnx=1,jpage
273 C
274 C SET LIMITS OF J TO BE PRINTED
275 C
276  iju(jnx) = j2 - (19*(jnx-1))
277  IF (jnx.NE.jpage) ijl(jnx) = j2 - (19*jnx)
278  IF (jnx.EQ.jpage) ijl(jnx) = j1
279  jmax = iju(jnx) - offj
280  jmin = ijl(jnx) - offj
281  ju = jjb - (4*jmax-3)
282  jl = jjb - (4*jmin-3)
283  nnn = (jmax-jmin)*4 + 1
284 C
285 C FILL CHARACTER ARRAY WITH BLANKS AND PUT X MARKERS IN CORNERS
286 C IF FIRST PANEL BLANK ENTIRE AREA,
287 C OTHERWISE TRANSFER FIRST INUM I BYTES TO LARGE ARRAY
288 C AND BLANK REMAINING ARRAY
289 C
290  DO 37 j=1,jjbb
291  IF (nx.NE.1) GO TO 31
292  DO 28 i=1,jjaapn
293  kh(i,j) = kb
294  28 CONTINUE
295  GO TO 37
296  31 CONTINUE
297  DO 32 i=1,lnump1
298  kh(i,j) = kk(i,j,jnx)
299  32 CONTINUE
300  DO 34 i=ibegin,jjaapn
301  kh(i,j) = kb
302  34 CONTINUE
303  37 CONTINUE
304  IF (jnx.NE.1) GO TO 40
305  kh(1,jjbb) = lc
306  kh(mmm,jjbb) = lc
307  200 CONTINUE
308  40 CONTINUE
309  IF (jnx.NE.jpage) GO TO 50
310  kh(1,1) = lc
311  kh(mmm,1) = lc
312  50 CONTINUE
313 C
314 C LOOP TO PUT DATA IN CHARACTER ARRAY
315 C
316  DO 110 i=1,idim
317 C
318 C TEST FOR BAD GEOGRAPHY
319 C
320  IF ((abs(alat(i)).GT.90.).OR.(alon(i).LT.0.0).OR.(alon
321  a (i).GT.360.0)) GO TO 90
322 C
323 C CHANGE LAT,LON TO I,J
324 C
325  IF (nref.EQ.0) GO TO 51
326  CALL w3fb04(alat(i),alon(i),-xmesh,260.0,deli,delj)
327  GO TO 52
328  51 CONTINUE
329  CALL w3fb04(alat(i),alon(i),xmesh,80.0,deli,delj)
330  52 CONTINUE
331  xi = xip + deli
332  xj = xjp + delj
333 C
334 C POSITION I,J COORDINATES IN CHARACTER ARRAY AND TEST
335 C IF VALUES RETURNED ARE WITHIN LIMITS OF MAP AND WITHIN SECTIONS
336 C
337  ii = 1.0 + (xi-offi-0.9001)*5.0
338  jj = 1.0 + (xj-offj-0.8751)*4.0
339  iw = (jjaam1*(nx-1)) + 1
340  ix = (jjaam1*nx) + 1
341  iy = jjb - (jjbbm1*(jnx-1))
342  IF (jnx.NE.jpage) iz = jjb - (jjbbm1*jnx)
343  IF (jnx.EQ.jpage) iz = 1
344  IF ((ii.LT.1).OR.(ii.GT.jja)) GO TO 100
345  IF ((jj.LT.1).OR.(jj.GT.jjb)) GO TO 100
346  IF ((ii.LT.iw).OR.(ii.GT.ix)) GO TO 100
347  IF ((jj.GT.iy).OR.(jj.LT.iz)) GO TO 100
348 C
349 C WRITE N+POLE IF IN THIS SECTION
350 C
351  IF (.NOT.((iixip.GE.iw.AND.iixip.LE.ix).AND.
352  a (jjxjp.LE.iy.AND.jjxjp.GE.iz))) GO TO 56
353  iixxp = iixip - (jjaam1*(nx-1))
354  jjxxp = jjxjp - (iz-1)
355  IF (nref.EQ.0) kh(iixxp-1,jjxxp) = kn
356  IF (nref.EQ.1) kh(iixxp-1,jjxxp) = ks
357  kh(iixxp,jjxxp) = kp
358  DO 53 l=1,4
359  kh(iixxp+l,jjxxp) = ipole(l)
360  53 CONTINUE
361  56 CONTINUE
362 C
363 C CONVERT CHARACTER ARRAY COORDINATES FROM
364 C TOTAL MAP VALUES TO SECTION VALUES
365 C
366  ii = ii - (jjaam1*(nx-1))
367  IF (jnx.NE.jpage) jj = jj - (iz-1)
368 C
369 C IF SPACE IS OCCUPIED SKIP THIS STATION
370 C
371  jnum = lnum + 1
372  DO 70 ik=1,jnum
373  in = ik - 1
374  IF (kh(ii+in,jj).EQ.kb) GO TO 60
375  GO TO 110
376  60 CONTINUE
377  70 CONTINUE
378 C
379 C PLACE VALUE TO BE PLOTTED IN CHARACTER ARRAY
380 C
381  ifield = ifld(i)
382 C
383 C TEST FOR TYPE OF DATA
384 C
385  IF ((inum/100).EQ.3) GO TO 82
386  IF ((inum/100).EQ.1) GO TO 73
387  jfld = rfield
388  GO TO 76
389  73 CONTINUE
390  jfld = ifield
391  76 CONTINUE
392 C
393 C IF ORIGINALLY FIXED POINT OR HAS BEEN CONVERTED
394 C FROM FLOATING POINT TO FIXED POINT
395 C
396  IF ((jfld/10000).GE.1) jfld = mod(jfld,10000)
397  iiabs = iabs(jfld)
398  CALL w3ai15(iiabs,imean,1,lnum,kp)
399  IF (jfld.LT.0) kh(ii,jj) = km
400  IF (jfld.GE.0) kh(ii,jj) = kp
401  DO 79 ia=1,lnum
402  kh(ii+ia,jj) = mean(ia)
403  79 CONTINUE
404  GO TO 110
405  82 CONTINUE
406 C
407 C FOR ALPHANUMERIC DATA
408 C
409  lfld = ifld(i)
410  kh(ii,jj) = kp
411  DO 85 iq=1,lnum
412  kh(ii+iq,jj) = DATA(iq)
413  85 CONTINUE
414  90 CONTINUE
415  100 CONTINUE
416  110 CONTINUE
417  jjn = 0
418 C
419 C PRINT JTH ROW AND VALUES OF J
420 C
421  DO 130 j=ju,jl,4
422  jn = nnn - (4*jjn)
423  IF (a) GO TO 115
424  jx = (jjb-j)/4 + 1
425  WRITE(6,fmt2) jx, (kh(i,jn), i=1,mmm), jx
426  115 CONTINUE
427  jjn = jjn + 1
428  IF (jn.NE.1) GO TO 118
429 C
430 C SAVE LAST INUM BYTES OF I
431 C
432  DO 117 l=1,jjbb
433  DO 116 i=116,jjaapn
434  ia = i - 115
435  kk(ia,l,jnx) = kh(i,l)
436  116 CONTINUE
437  117 CONTINUE
438  a = .true.
439  GO TO 140
440  118 CONTINUE
441  DO 120 im=1,3
442  jn = jn - 1
443  print 1003, (kh(i,jn), i=1,mmm)
444  120 CONTINUE
445  a = .false.
446  130 CONTINUE
447  140 CONTINUE
448  WRITE(6,fmt4) (line(n), n=1,la)
449 C
450 C CALCULATE AND PRINT LAT/LON AT CORNERS
451 C
452  al = il(nx)
453  ar = ir(nx)
454  xi1 = ((al-1.0)/xfac + 1.0) - 33.0
455  xi2 = ((ar-1.0)/xfac + 1.0) - 33.0
456  xj1 = cmjb - 33.0
457  xj2 = cmjt - 33.0
458  IF (nref.EQ.0) GO TO 142
459  CALL w3fb05(xi1,xj1,-xmesh,260.0,alat1,alon1)
460  CALL w3fb05(xi1,xj2,-xmesh,260.0,alat2,alon2)
461  CALL w3fb05(xi2,xj2,-xmesh,260.0,alat3,alon3)
462  CALL w3fb05(xi2,xj1,-xmesh,260.0,alat4,alon4)
463  GO TO 144
464  142 CONTINUE
465  CALL w3fb05(xi1,xj1,xmesh,80.0,alat1,alon1)
466  CALL w3fb05(xi1,xj2,xmesh,80.0,alat2,alon2)
467  CALL w3fb05(xi2,xj2,xmesh,80.0,alat3,alon3)
468  CALL w3fb05(xi2,xj1,xmesh,80.0,alat4,alon4)
469  144 CONTINUE
470  print 2001, alat2, alon2, alat3, alon3
471  print 2002, alat1, alon1, alat4, alon4
472  print 2003, nx, ipage, title
473  150 CONTINUE
474  ierr = 0
475  RETURN
476  END
