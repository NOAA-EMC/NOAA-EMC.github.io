6 C$$$ SUBPROGRAM DOCUMENTATION BLOCK
7 C
8 C SUBPROGRAM: SPTRAND PERFORM A GRADIENT SPHERICAL TRANSFORM
9 C PRGMMR: IREDELL ORG: W/NMC23 DATE: 96-02-29
10 C
11 C ABSTRACT: THIS SUBPROGRAM PERFORMS A SPHERICAL TRANSFORM
12 C BETWEEN SPECTRAL COEFFICIENTS OF SCALAR FIELDS
13 C AND THEIR MEANS AND GRADIENTS ON A GLOBAL CYLINDRICAL GRID.
14 C THE WAVE-SPACE CAN BE EITHER TRIANGULAR OR RHOMBOIDAL.
15 C THE GRID-SPACE CAN BE EITHER AN EQUALLY-SPACED GRID
16 C (WITH OR WITHOUT POLE POINTS) OR A GAUSSIAN GRID.
17 C THE WAVE AND GRID FIELDS MAY HAVE GENERAL INDEXING,
18 C BUT EACH WAVE FIELD IS IN SEQUENTIAL 'IBM ORDER',
19 C I.E. WITH ZONAL WAVENUMBER AS THE SLOWER INDEX.
20 C TRANSFORMS ARE DONE IN LATITUDE PAIRS FOR EFFICIENCY;
21 C THUS GRID ARRAYS FOR EACH HEMISPHERE MUST BE PASSED.
22 C IF SO REQUESTED, JUST A SUBSET OF THE LATITUDE PAIRS
23 C MAY BE TRANSFORMED IN EACH INVOCATION OF THE SUBPROGRAM.
24 C THE TRANSFORMS ARE ALL MULTIPROCESSED OVER LATITUDE EXCEPT
25 C THE TRANSFORM FROM FOURIER TO SPECTRAL IS MULTIPROCESSED
26 C OVER ZONAL WAVENUMBER TO ENSURE REPRODUCIBILITY.
27 C TRANSFORM SEVERAL FIELDS AT A TIME TO IMPROVE VECTORIZATION.
28 C SUBPROGRAM CAN BE CALLED FROM A MULTIPROCESSING ENVIRONMENT.
29 C
30 C PROGRAM HISTORY LOG:
31 C 96-02-29 IREDELL
32 C 1998-12-15 IREDELL OPENMP DIRECTIVES INSERTED
33 C
34 C USAGE: CALL SPTRAND(IROMB,MAXWV,IDRT,IMAX,JMAX,KMAX,
35 C & IPRIME,ISKIP,JNSKIP,JSSKIP,KWSKIP,KGSKIP,
36 C & JBEG,JEND,JCPU,
37 C & WAVE,GRIDMN,GRIDXN,GRIDXS,GRIDYN,GRIDYS,IDIR)
38 C INPUT ARGUMENTS:
39 C IROMB - INTEGER SPECTRAL DOMAIN SHAPE
40 C (0 FOR TRIANGULAR, 1 FOR RHOMBOIDAL)
41 C MAXWV - INTEGER SPECTRAL TRUNCATION
42 C IDRT - INTEGER GRID IDENTIFIER
43 C (IDRT=4 FOR GAUSSIAN GRID,
44 C IDRT=0 FOR EQUALLY-SPACED GRID INCLUDING POLES,
45 C IDRT=256 FOR EQUALLY-SPACED GRID EXCLUDING POLES)
46 C IMAX - INTEGER EVEN NUMBER OF LONGITUDES.
47 C JMAX - INTEGER NUMBER OF LATITUDES.
48 C KMAX - INTEGER NUMBER OF FIELDS TO TRANSFORM.
49 C IPRIME - INTEGER LONGITUDE INDEX FOR THE PRIME MERIDIAN.
50 C (DEFAULTS TO 1 IF IPRIME=0)
51 C ISKIP - INTEGER SKIP NUMBER BETWEEN LONGITUDES
52 C (DEFAULTS TO 1 IF ISKIP=0)
53 C JNSKIP - INTEGER SKIP NUMBER BETWEEN N.H. LATITUDES FROM NORTH
54 C (DEFAULTS TO IMAX IF JNSKIP=0)
55 C JSSKIP - INTEGER SKIP NUMBER BETWEEN S.H. LATITUDES FROM SOUTH
56 C (DEFAULTS TO -IMAX IF JSSKIP=0)
57 C KWSKIP - INTEGER SKIP NUMBER BETWEEN WAVE FIELDS
58 C (DEFAULTS TO (MAXWV+1)*((IROMB+1)*MAXWV+2) IF KWSKIP=0)
59 C KGSKIP - INTEGER SKIP NUMBER BETWEEN GRID FIELDS
60 C (DEFAULTS TO IMAX*JMAX IF KGSKIP=0)
61 C JBEG - INTEGER LATITUDE INDEX (FROM POLE) TO BEGIN TRANSFORM
62 C (DEFAULTS TO 1 IF JBEG=0)
63 C (IF JBEG=0 AND IDIR<0, WAVE IS ZEROED BEFORE TRANSFORM)
64 C JEND - INTEGER LATITUDE INDEX (FROM POLE) TO END TRANSFORM
65 C (DEFAULTS TO (JMAX+1)/2 IF JEND=0)
66 C JCPU - INTEGER NUMBER OF CPUS OVER WHICH TO MULTIPROCESS
67 C WAVE - REAL (*) WAVE FIELDS IF IDIR>0
68 C GRIDMN - REAL (KMAX) GLOBAL MEANS IF IDIR<0
69 C GRIDXN - REAL (*) N.H. X-GRADIENTS (STARTING AT JBEG) IF IDIR<0
70 C GRIDXS - REAL (*) S.H. X-GRADIENTS (STARTING AT JBEG) IF IDIR<0
71 C GRIDYN - REAL (*) N.H. Y-GRADIENTS (STARTING AT JBEG) IF IDIR<0
72 C GRIDYS - REAL (*) S.H. Y-GRADIENTS (STARTING AT JBEG) IF IDIR<0
73 C IDIR - INTEGER TRANSFORM FLAG
74 C (IDIR>0 FOR WAVE TO GRID, IDIR<0 FOR GRID TO WAVE)
75 C OUTPUT ARGUMENTS:
76 C WAVE - REAL (*) WAVE FIELDS IF IDIR<0
77 C GRIDMN - REAL (KMAX) GLOBAL MEANS IF IDIR>0
78 C GRIDXN - REAL (*) N.H. X-GRADIENTS (STARTING AT JBEG) IF IDIR>0
79 C GRIDXS - REAL (*) S.H. X-GRADIENTS (STARTING AT JBEG) IF IDIR>0
80 C [GRIDX=(D(WAVE)/DLAM)/(CLAT*RERTH)]
81 C GRIDYN - REAL (*) N.H. Y-GRADIENTS (STARTING AT JBEG) IF IDIR>0
82 C GRIDYS - REAL (*) S.H. Y-GRADIENTS (STARTING AT JBEG) IF IDIR>0
83 C [GRIDY=(D(WAVE)/DPHI)/RERTH]
84 C
85 C SUBPROGRAMS CALLED:
86 C SPWGET GET WAVE-SPACE CONSTANTS
87 C SPLAPLAC COMPUTE LAPLACIAN IN SPECTRAL SPACE
88 C SPTRANV PERFORM A VECTOR SPHERICAL TRANSFORM
89 C
90 C REMARKS: MINIMUM GRID DIMENSIONS FOR UNALIASED TRANSFORMS TO SPECTRAL:
91 C DIMENSION LINEAR QUADRATIC
92 C ----------------------- --------- -------------
93 C IMAX 2*MAXWV+2 3*MAXWV/2*2+2
94 C JMAX (IDRT=4,IROMB=0) 1*MAXWV+1 3*MAXWV/2+1
95 C JMAX (IDRT=4,IROMB=1) 2*MAXWV+1 5*MAXWV/2+1
96 C JMAX (IDRT=0,IROMB=0) 2*MAXWV+3 3*MAXWV/2*2+3
97 C JMAX (IDRT=0,IROMB=1) 4*MAXWV+3 5*MAXWV/2*2+3
98 C JMAX (IDRT=256,IROMB=0) 2*MAXWV+1 3*MAXWV/2*2+1
99 C JMAX (IDRT=256,IROMB=1) 4*MAXWV+1 5*MAXWV/2*2+1
100 C ----------------------- --------- -------------
101 C
102 C ATTRIBUTES:
103 C LANGUAGE: FORTRAN 77
104 C
105 C$$$
106  REAL WAVE(*),GRIDMN(KMAX),GRIDXN(*),GRIDXS(*),GRIDYN(*),GRIDYS(*)
107  REAL EPS((MAXWV+1)*((IROMB+1)*MAXWV+2)/2),EPSTOP(MAXWV+1)
108  REAL ENN1((MAXWV+1)*((IROMB+1)*MAXWV+2)/2)
109  REAL ELONN1((MAXWV+1)*((IROMB+1)*MAXWV+2)/2)

113 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
114 C SET PARAMETERS
115  CALL spwget(iromb,maxwv,eps,epstop,enn1,elonn1,eon,eontop)
116  mx=(maxwv+1)*((iromb+1)*maxwv+2)/2
117  mdim=2*mx+1
118  kw=kwskip

124  DO k=1,kmax

128  wz(1:2*mx,k)=0.

131  & iprime,iskip,jnskip,jsskip,mdim,kgskip,
132  & jbeg,jend,jcpu,
133  & wd,wz,gridxn,gridxs,gridyn,gridys,idir)
134 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
135 C TRANSFORM GRID TO WAVE
136  ELSE
137 C$OMP PARALLEL DO
138  DO k=1,kmax
139  wd(1:2*mx,k)=0.
140  wz(1:2*mx,k)=0.
141  ENDDO
142  CALL sptranv(iromb,maxwv,idrt,imax,jmax,kmax,
143  & iprime,iskip,jnskip,jsskip,mdim,kgskip,
144  & jbeg,jend,jcpu,
145  & wd,wz,gridxn,gridxs,gridyn,gridys,idir)
146  IF(jbeg.EQ.0) THEN
147 C$OMP PARALLEL DO PRIVATE(KWS)
148  DO k=1,kmax
149  kws=(k-1)*kw
150  CALL splaplac(iromb,maxwv,enn1,wave(kws+1),wd(1,k),-1)
151  wave(kws+1)=gridmn(k)*sqrt(2.)
152  ENDDO
153  ELSE
154 C$OMP PARALLEL DO PRIVATE(KWS)
155  DO k=1,kmax
156  kws=(k-1)*kw
157  CALL splaplac(iromb,maxwv,enn1,wz(1,k),wd(1,k),-1)
158  wave(kws+1:kws+2*mx)=wave(kws+1:kws+2*mx)+wz(1:2*mx,k)
159  wave(kws+1)=gridmn(k)*sqrt(2.)
160  ENDDO
161  ENDIF
162  ENDIF
163 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
164  END
