2  SUBROUTINE splegend(I,M,SLAT,CLAT,EPS,EPSTOP,PLN,PLNTOP)
3 C$$$ SUBPROGRAM DOCUMENTATION BLOCK
4 C
5 C SUBPROGRAM: SPLEGEND COMPUTE LEGENDRE POLYNOMIALS
6 C PRGMMR: IREDELL ORG: W/NMC23 DATE: 92-10-31
7 C
8 C ABSTRACT: EVALUATES THE ORTHONORMAL ASSOCIATED LEGENDRE POLYNOMIALS
9 C IN THE SPECTRAL DOMAIN AT A GIVEN LATITUDE.
10 C SUBPROGRAM SPLEGEND SHOULD BE CALLED ALREADY.
11 C IF L IS THE ZONAL WAVENUMBER, N IS THE TOTAL WAVENUMBER,
12 C AND EPS(L,N)=SQRT((N**2-L**2)/(4*N**2-1)) THEN
13 C THE FOLLOWING BOOTSTRAPPING FORMULAS ARE USED:
14 C PLN(0,0)=SQRT(0.5)
15 C PLN(L,L)=PLN(L-1,L-1)*CLAT*SQRT(FLOAT(2*L+1)/FLOAT(2*L))
16 C PLN(L,N)=(SLAT*PLN(L,N-1)-EPS(L,N-1)*PLN(L,N-2))/EPS(L,N)
17 C SYNTHESIS AT THE POLE NEEDS ONLY TWO ZONAL WAVENUMBERS.
18 C SCALAR FIELDS ARE SYNTHESIZED WITH ZONAL WAVENUMBER 0 WHILE
19 C VECTOR FIELDS ARE SYNTHESIZED WITH ZONAL WAVENUMBER 1.
20 C (THUS POLAR VECTOR FIELDS ARE IMPLICITLY DIVIDED BY CLAT.)
21 C THE FOLLOWING BOOTSTRAPPING FORMULAS ARE USED AT THE POLE:
22 C PLN(0,0)=SQRT(0.5)
23 C PLN(1,1)=SQRT(0.75)
24 C PLN(L,N)=(PLN(L,N-1)-EPS(L,N-1)*PLN(L,N-2))/EPS(L,N)
25 C
26 C PROGRAM HISTORY LOG:
27 C 91-10-31 MARK IREDELL
28 C 98-06-10 MARK IREDELL GENERALIZE PRECISION
29 C
30 C USAGE: CALL SPLEGEND(I,M,SLAT,CLAT,EPS,EPSTOP,PLN,PLNTOP)
31 C
32 C INPUT ARGUMENT LIST:
33 C I - INTEGER SPECTRAL DOMAIN SHAPE
34 C (0 FOR TRIANGULAR, 1 FOR RHOMBOIDAL)
35 C M - INTEGER SPECTRAL TRUNCATION
36 C SLAT - REAL SINE OF LATITUDE
37 C CLAT - REAL COSINE OF LATITUDE
38 C EPS - REAL ((M+1)*((I+1)*M+2)/2) SQRT((N**2-L**2)/(4*N**2-1))
39 C EPSTOP - REAL (M+1) SQRT((N**2-L**2)/(4*N**2-1)) OVER TOP
40 C
41 C OUTPUT ARGUMENT LIST:
42 C PLN - REAL ((M+1)*((I+1)*M+2)/2) LEGENDRE POLYNOMIAL
43 C PLNTOP - REAL (M+1) LEGENDRE POLYNOMIAL OVER TOP
44 C
45 C ATTRIBUTES:
46 C LANGUAGE: FORTRAN 90
47 C
48 C$$$
49 CFPP$ NOCONCUR R
50  REAL EPS((M+1)*((I+1)*M+2)/2),EPSTOP(M+1)
51  REAL PLN((M+1)*((I+1)*M+2)/2),PLNTOP(M+1)
52  REAL(KIND=selected_real_kind(15,45)):: dln((m+1)*((i+1)*m+2)/2)
53 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
54 C ITERATIVELY COMPUTE PLN WITHIN SPECTRAL DOMAIN AT POLE
55  m1=m+1
56  m2=2*m+i+1
57  mx=(m+1)*((i+1)*m+2)/2
58  IF(clat.EQ.0.) THEN
59  dln(1)=sqrt(0.5)
60  IF(m.GT.0) THEN
61  dln(m1+1)=sqrt(0.75)
62  dln(2)=slat*dln(1)/eps(2)
63  ENDIF
64  IF(m.GT.1) THEN
65  dln(m1+2)=slat*dln(m1+1)/eps(m1+2)
66  dln(3)=(slat*dln(2)-eps(2)*dln(1))/eps(3)
67  DO n=3,m
68  k=1+n
69  dln(k)=(slat*dln(k-1)-eps(k-1)*dln(k-2))/eps(k)
70  k=m1+n
71  dln(k)=(slat*dln(k-1)-eps(k-1)*dln(k-2))/eps(k)
72  ENDDO
73  IF(i.EQ.1) THEN
74  k=m2
75  dln(k)=(slat*dln(k-1)-eps(k-1)*dln(k-2))/eps(k)
76  ENDIF
77  DO k=m2+1,mx
78  dln(k)=0.
79  ENDDO
80  ENDIF
81 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
82 C COMPUTE POLYNOMIALS OVER TOP OF SPECTRAL DOMAIN
83  k=m1+1
84  plntop(1)=(slat*dln(k-1)-eps(k-1)*dln(k-2))/epstop(1)
85  IF(m.GT.0) THEN
86  k=m2+1
87  plntop(2)=(slat*dln(k-1)-eps(k-1)*dln(k-2))/epstop(2)
88  DO l=2,m
89  plntop(l+1)=0.
90  ENDDO
91  ENDIF
92 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
93 C ITERATIVELY COMPUTE PLN(L,L) (BOTTOM HYPOTENUSE OF DOMAIN)
94  ELSE
95  nml=0
96  k=1
97  dln(k)=sqrt(0.5)
98  DO l=1,m+(i-1)*nml
99  kp=k
100  k=l*(2*m+(i-1)*(l-1))/2+l+nml+1
101  dln(k)=dln(kp)*clat*sqrt(float(2*l+1)/float(2*l))
102  ENDDO
103 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
104 C COMPUTE PLN(L,L+1) (DIAGONAL NEXT TO BOTTOM HYPOTENUSE OF DOMAIN)
105  nml=1
106 CDIR$ IVDEP
107  DO l=0,m+(i-1)*nml
108  k=l*(2*m+(i-1)*(l-1))/2+l+nml+1
109  dln(k)=slat*dln(k-1)/eps(k)
110  ENDDO
111 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
112 C COMPUTE REMAINING PLN IN SPECTRAL DOMAIN
113  DO nml=2,m
114 CDIR$ IVDEP
115  DO l=0,m+(i-1)*nml
116  k=l*(2*m+(i-1)*(l-1))/2+l+nml+1
117  dln(k)=(slat*dln(k-1)-eps(k-1)*dln(k-2))/eps(k)
118  ENDDO
119  ENDDO
120 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
121 C COMPUTE POLYNOMIALS OVER TOP OF SPECTRAL DOMAIN
122  DO l=0,m
123  nml=m+1+(i-1)*l
124  k=l*(2*m+(i-1)*(l-1))/2+l+nml+1
125  plntop(l+1)=(slat*dln(k-1)-eps(k-1)*dln(k-2))/epstop(l+1)
126  ENDDO
127  ENDIF
128 C - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
129 C RETURN VALUES
130  DO k=1,mx
131  pln(k)=dln(k)
132  ENDDO
133  RETURN
134  END
