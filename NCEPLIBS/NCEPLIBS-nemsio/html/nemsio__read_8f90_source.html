2 module nemsio_read
3 !
4 !$$$ documentation clock
5 !
6 ! module: nemsio_read read data fields from a nemsio file
7 ! Programmer: J. Wang date: 2011-01-13
8 !
9 ! abstract: this module provides subroutines to read data fields out of a
10 ! nemsio file. The data file could be 'bin4','bin8', or 'grib',
11 ! The dat field could be read out by the record number
12 ! or by given the data field name, level type and level. Overload
13 ! interfaces are provided to handle different data type (real(4)
14 ! or real(8) ) of the array that holds the data.
15 !
16 ! Possible return code
17 ! 0 Successful call
18 ! -31 get dimension ffrom gfile
19 ! -32 read data field by record number using w3d
20 ! -33 read data field by given data field name,level type and level using w3d
21 ! -34 read data field by record number using w34
22 ! -35 read data field by given data field name,level type and level using w34
23 ! -41 read data field by record number from 4 byte real binary file
24 ! -42 read data field by record number from 8 byte real binary file
25 ! -43 read data field by field name,levtyp and lev from 4 byte real binary file
26 ! -44 read data field by field name,levtyp and lev from 8 byte real binary file
27 ! -45 read data field by record number using w34 from grib data
28 ! -46 read data field by field name,level type and level using w34 from grib data
29 ! -47 read data field by record number using w3d from grib data
30 ! -48 read data field by field name,level type and level using w3d from grib data
31 !------------------------------------------------------------------------------

128  implicit none

133  integer(nemsio_intkind),optional,intent(in) :: nframe

135  real(nemsio_dblekind),allocatable :: datatmp8(:)
136  integer :: i,j,ios

140  if(present(iret)) iret=-32

144  if ( mygdatatype .eq. 'bin4') then

147  else
148  allocate(datatmp(myfieldsize) )
149  call nemsio_readrecbin4d4(gfile,jrec,datatmp,ios)
150  endif
151  else if ( mygdatatype .eq. 'bin8') then

162  else
163  call nemsio_stop
164  endif
165  endif
166 !---
167  if ( present(nframe) ) then
168  if(mygdatatype .eq. 'bin4') then
169  do j=1,mydimy+2*mynframe-2*nframe
170  do i=1,mydimx+2*mynframe -2*nframe
171  data(i+(j-1)*(mydimx+2*mynframe-2*nframe))=datatmp(i+nframe &
172  +(j-1+nframe)*(mydimx+2*mynframe))
173  enddo
174  enddo
175  deallocate(datatmp)
176  elseif(mygdatatype=='bin8'.or.mygdatatype=='grib') then
177  do j=1,mydimy+2*mynframe-2*nframe
178  do i=1,mydimx+2*mynframe -2*nframe
179  data(i+(j-1)*(mydimx+2*mynframe-2*nframe))=datatmp8(i+nframe &
180  +(j-1+nframe)*(mydimx+2*mynframe))
181  enddo
182  enddo
183  deallocate(datatmp8)
184  endif
185  else
186  if(mygdatatype=='bin8'.or.mygdatatype=='grib') then
187  data=datatmp8
188  deallocate(datatmp8)
189  endif
190  endif
191 !---
192  if(present(iret)) iret=0
193  return
194  end subroutine nemsio_readrec4
