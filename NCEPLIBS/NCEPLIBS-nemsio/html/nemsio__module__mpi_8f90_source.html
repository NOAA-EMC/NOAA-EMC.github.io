371  if (present(iret))iret=1
372 !------------------------------------------------------------
373 ! MPI set meta data type

376  meta1_type(1)=mpi_character
377  meta1_type(2)=mpi_integer
378  meta1_block(1)=24
379  meta1_block(2)=6

384  call mpi_type_commit(itypemeta1,ios)
385  if ( ios.ne.0 ) then
386  return
387  endif
388 !

390  meta2_type(1)=mpi_integer
391  meta2_type(2)=mpi_real
392  meta2_type(3)=mpi_logical

395  meta2_block(3)=1
396  meta2_disp(1)=0
397  meta2_disp(2)=meta2_block(1)*4+meta2_disp(1)
398  meta2_disp(3)=meta2_block(2)*4+meta2_disp(2)
399  call mpi_type_struct(3,meta2_block,meta2_disp,meta2_type, &
400  itypemeta2,ios)
401  call mpi_type_commit(itypemeta2,ios)
402  if ( ios.ne.0 ) then
403  return
404  endif
405 !
406 !------------------------------------------------------------
407 ! check machine endian
408 !------------------------------------------------------------
409  call chk_endianc(machine_endian)
410  if(trim(machine_endian)=='mixed_endian') then
411  call nemsio_stop('You are in mixed endian computer,stop!!!')
412  endif
413 !
414  if(present(iret)) iret=0
